36      @Test
37      void testConstructors() throws WrongSizeException {
38          // Test empty constructor
39          var params = new AffineParameters3D();
40  
41          // check correctness
42          assertEquals(AffineParameters3D.DEFAULT_SCALE, params.getScaleX(), 0.0);
43          assertEquals(AffineParameters3D.DEFAULT_SCALE, params.getScaleY(), 0.0);
44          assertEquals(AffineParameters3D.DEFAULT_SCALE, params.getScaleZ(), 0.0);
45          assertEquals(AffineParameters3D.DEFAULT_SKEWNESS, params.getSkewnessXY(), 0.0);
46          assertEquals(AffineParameters3D.DEFAULT_SKEWNESS, params.getSkewnessXZ(), 0.0);
47          assertEquals(AffineParameters3D.DEFAULT_SKEWNESS, params.getSkewnessYZ(), 0.0);
48  
49          // Test constructor with scale
50          final var randomizer = new UniformRandomizer();
51          final var scale = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
52  
53          params = new AffineParameters3D(scale);
54  
55          // check correctness
56          assertEquals(scale, params.getScaleX(), 0.0);
57          assertEquals(scale, params.getScaleY(), 0.0);
58          assertEquals(scale, params.getScaleZ(), 0.0);
59          assertEquals(AffineParameters3D.DEFAULT_SKEWNESS, params.getSkewnessXY(), 0.0);
60          assertEquals(AffineParameters3D.DEFAULT_SKEWNESS, params.getSkewnessXZ(), 0.0);
61          assertEquals(AffineParameters3D.DEFAULT_SKEWNESS, params.getSkewnessYZ(), 0.0);
62  
63          // Test constructor with scale and skewness
64          final var skewness = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
65  
66          params = new AffineParameters3D(scale, skewness);
67  
68          // check correctness
69          assertEquals(scale, params.getScaleX(), 0.0);
70          assertEquals(scale, params.getScaleY(), 0.0);
71          assertEquals(scale, params.getScaleZ(), 0.0);
72          assertEquals(skewness, params.getSkewnessXY(), 0.0);
73          assertEquals(skewness, params.getSkewnessXZ(), 0.0);
74          assertEquals(skewness, params.getSkewnessYZ(), 0.0);
75  
76          // Test constructor with scaleX, scaleY, scaleZ, skewnessXY, skewnessXZ
77          // and skewnessYZ
78          final var scaleX = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
79          final var scaleY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
80          final var scaleZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
81          final var skewnessXY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
82          final var skewnessXZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
83          final var skewnessYZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
84  
85          params = new AffineParameters3D(scaleX, scaleY, scaleZ, skewnessXY, skewnessXZ, skewnessYZ);
86  
87          // check correctness
88          assertEquals(scaleX, params.getScaleX(), 0.0);
89          assertEquals(scaleY, params.getScaleY(), 0.0);
90          assertEquals(scaleZ, params.getScaleZ(), 0.0);
91          assertEquals(skewnessXY, params.getSkewnessXY(), 0.0);
92          assertEquals(skewnessXZ, params.getSkewnessXZ(), 0.0);
93          assertEquals(skewnessYZ, params.getSkewnessYZ(), 0.0);
94  
95          // test constructor with matrix
96          final var m = new Matrix(INHOM_COORDS, INHOM_COORDS);
97          m.setElementAt(0, 0, scaleX);
98          m.setElementAt(1, 1, scaleY);
99          m.setElementAt(2, 2, scaleZ);
100         m.setElementAt(0, 1, skewnessXY);
101         m.setElementAt(0, 2, skewnessXZ);
102         m.setElementAt(1, 2, skewnessYZ);
103 
104         params = new AffineParameters3D(m);
105 
106         // check correctness
107         assertEquals(scaleX, params.getScaleX(), 0.0);
108         assertEquals(scaleY, params.getScaleY(), 0.0);
109         assertEquals(scaleZ, params.getScaleZ(),  0.0);
110         assertEquals(skewnessXY, params.getSkewnessXY(), 0.0);
111         assertEquals(skewnessXZ, params.getSkewnessXZ(), 0.0);
112         assertEquals(skewnessYZ, params.getSkewnessYZ(), 0.0);
113 
114         // Force IllegalArgumentException (invalid size)
115         final var badM1 = new Matrix(INHOM_COORDS + 1, INHOM_COORDS + 1);
116         assertThrows(IllegalArgumentException.class, () -> new AffineParameters3D(badM1));
117 
118         // Force IllegalArgumentException (non upper triangular matrix
119         final var badM2 = new Matrix(INHOM_COORDS, INHOM_COORDS);
120         badM2.setElementAt(1, 0, 1.0);
121         assertThrows(IllegalArgumentException.class, () -> new AffineParameters3D(badM2));
122 
123         // Test constructor with matrix and threshold
124         final var threshold = randomizer.nextDouble();
125         params = new AffineParameters3D(m, threshold);
126 
127         // check correctness
128         assertEquals(scaleX, params.getScaleX(), 0.0);
129         assertEquals(scaleY, params.getScaleY(), 0.0);
130         assertEquals(scaleZ, params.getScaleZ(), 0.0);
131         assertEquals(skewnessXY, params.getSkewnessXY(), 0.0);
132         assertEquals(skewnessXZ, params.getSkewnessXZ(), 0.0);
133         assertEquals(skewnessYZ, params.getSkewnessYZ(), 0.0);
134 
135         // Force IllegalArgumentException (invalid size)
136         assertThrows(IllegalArgumentException.class, () -> new AffineParameters3D(badM1, threshold));
137 
138         // Force IllegalArgumentException (non upper triangular matrix)
139         badM2.setElementAt(1, 0, threshold + 1.0);
140         assertThrows(IllegalArgumentException.class, () -> new AffineParameters3D(badM2));
141 
142         // Force IllegalArgumentException (with negative threshold)
143         assertThrows(IllegalArgumentException.class, () -> new AffineParameters3D(m, -threshold));
144     }

269     @Test
270     void testAsMatrix() throws WrongSizeException {
271         final var randomizer = new UniformRandomizer();
272         final double scaleX = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
273         final var scaleY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
274         final var scaleZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
275         final var skewnessXY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
276         final var skewnessXZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
277         final var skewnessYZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
278 
279         final var params = new AffineParameters3D(scaleX, scaleY, scaleZ, skewnessXY, skewnessXZ, skewnessYZ);
280 
281         final var expectedMatrix = new Matrix(INHOM_COORDS, INHOM_COORDS);
282         expectedMatrix.initialize(0.0);
283         expectedMatrix.setElementAt(0, 0, scaleX);
284         expectedMatrix.setElementAt(1, 1, scaleY);
285         expectedMatrix.setElementAt(2, 2, scaleZ);
286         expectedMatrix.setElementAt(0, 1, skewnessXY);
287         expectedMatrix.setElementAt(0, 2, skewnessXZ);
288         expectedMatrix.setElementAt(1, 2, skewnessYZ);
289 
290         final var m1 = params.asMatrix();
291         final var m2 = new Matrix(INHOM_COORDS, INHOM_COORDS);
292         params.asMatrix(m2);
293 
294         assertTrue(expectedMatrix.equals(m1, ABSOLUTE_ERROR));
295         assertTrue(expectedMatrix.equals(m2, ABSOLUTE_ERROR));
296     }

298     @Test
299     void testFromMatrixAndIsValidMatrix() throws WrongSizeException {
300         final var randomizer = new UniformRandomizer();
301         final var scaleX = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
302         final var scaleY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
303         final var scaleZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
304         final var skewnessXY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
305         final var skewnessXZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
306         final var skewnessYZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
307         final var threshold = randomizer.nextDouble();
308 
309         // build valid matrix
310         final var m = new Matrix(INHOM_COORDS, INHOM_COORDS);
311         m.setElementAt(0, 0, scaleX);
312         m.setElementAt(1, 1, scaleY);
313         m.setElementAt(2, 2, scaleZ);
314         m.setElementAt(0, 1, skewnessXY);
315         m.setElementAt(0, 2, skewnessXZ);
316         m.setElementAt(1, 2, skewnessYZ);
317 
318         // check matrix is valid
319         assertTrue(AffineParameters3D.isValidMatrix(m));
320         assertTrue(AffineParameters3D.isValidMatrix(m, threshold));
321 
322         // Force IllegalArgumentException when checking validity (negative
323         // threshold)
324         assertThrows(IllegalArgumentException.class, () -> AffineParameters3D.isValidMatrix(m, -threshold));
325 
326         final var params = new AffineParameters3D();
327 
328         // set parameters from matrix
329         params.fromMatrix(m);
330 
331         // check correctness
332         assertEquals(scaleX, params.getScaleX(), 0.0);
333         assertEquals(scaleY, params.getScaleY(), 0.0);
334         assertEquals(scaleZ, params.getScaleZ(), 0.0);
335         assertEquals(skewnessXY, params.getSkewnessXY(), 0.0);
336         assertEquals(skewnessXZ, params.getSkewnessXZ(), 0.0);
337         assertEquals(skewnessYZ, params.getSkewnessYZ(), 0.0);
338 
339         // Force IllegalArgumentException (invalid size)
340         final var badM1 = new Matrix(INHOM_COORDS + 1, INHOM_COORDS + 1);
341         assertThrows(IllegalArgumentException.class, () -> params.fromMatrix(badM1));
342 
343         // Force IllegalArgumentException (non upper triangular matrix
344         final var badM2 = new Matrix(INHOM_COORDS, INHOM_COORDS);
345         badM2.setElementAt(1, 0, 1.0);
346         assertThrows(IllegalArgumentException.class, () -> params.fromMatrix(badM2));
347 
348         // Test from matrix with threshold
349         params.fromMatrix(m, threshold);
350 
351         // check correctness
352         assertEquals(scaleX, params.getScaleX(), 0.0);
353         assertEquals(scaleY, params.getScaleY(), 0.0);
354         assertEquals(scaleZ, params.getScaleZ(), 0.0);
355         assertEquals(skewnessXY, params.getSkewnessXY(), 0.0);
356         assertEquals(skewnessXZ, params.getSkewnessXZ(), 0.0);
357         assertEquals(skewnessYZ, params.getSkewnessYZ(), 0.0);
358 
359         // Force IllegalArgumentException (invalid size)
360         assertThrows(IllegalArgumentException.class, () -> params.fromMatrix(badM1, threshold));
361 
362         // Force IllegalArgumentException (non upper triangular matrix
363         badM2.setElementAt(1, 0, threshold + 1.0);
364         assertThrows(IllegalArgumentException.class, () -> params.fromMatrix(badM2, threshold));
365 
366         // Force IllegalArgumentException (with negative threshold)
367         assertThrows(IllegalArgumentException.class, () -> params.fromMatrix(m, -threshold));
368     }

372         final var randomizer = new UniformRandomizer();

379 
380         final var params1 = new AffineParameters3D(scaleX, scaleY, scaleZ, skewnessXY, skewnessXZ, skewnessYZ);
