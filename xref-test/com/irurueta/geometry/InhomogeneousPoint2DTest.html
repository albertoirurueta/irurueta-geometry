28      private static final int HOM_COORDS = 3;
29      private static final int INHOM_COORDS = 2;

31      private static final double ABSOLUTE_ERROR = 1e-6;
32      private static final double MIN_RANDOM_VALUE = 1.0;

35      @Test
36      void testConstructor() {
37          var iPoint = new InhomogeneousPoint2D();
38          assertEquals(0.0, iPoint.getInhomX(), 0.0);
39          assertEquals(0.0, iPoint.getInhomY(), 0.0);
40  
41          final var randomizer = new UniformRandomizer();
42          var array = new double[INHOM_COORDS];
43          randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
44  
45          iPoint = new InhomogeneousPoint2D(array);
46  
47          final var array2 = iPoint.asArray();
48  
49          assertArrayEquals(array, array2, 0.0);
50  
51          var a = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
52          var b = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
53  
54          iPoint = new InhomogeneousPoint2D(a, b);
55          array = iPoint.asArray();
56  
57          assertEquals(a, array[0], 0.0);
58          assertEquals(b, array[1], 0.0);
59  
60          final var point = Point2D.create(CoordinatesType.INHOMOGENEOUS_COORDINATES);
61          a = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
62          b = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
63          point.setInhomogeneousCoordinates(a, b);
64          iPoint = new InhomogeneousPoint2D(point);
65          assertEquals(iPoint.getInhomX(), a, 0.0);
66          assertEquals(iPoint.getInhomY(), b, 0.0);
67          assertEquals(CoordinatesType.INHOMOGENEOUS_COORDINATES, iPoint.getType());
68      }

70      @Test
71      void testGettersAndSetters() {
72          final var randomizer = new UniformRandomizer();
73          final var x = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
74          final var y = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
75          final var homX = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
76          final var homY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
77          final var homW = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
78          final var inhomX = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
79          final var inhomY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
80  
81          final var iPoint = new InhomogeneousPoint2D();
82  
83          iPoint.setX(x);
84          iPoint.setY(y);
85          assertEquals(iPoint.getX(), x, 0.0);
86          assertEquals(iPoint.getY(), y, 0.0);
87  
88          iPoint.setHomogeneousCoordinates(homX, homY, homW);
89          final var constantX = iPoint.getHomX() / homX;
90          final var constantY = iPoint.getHomY() / homY;
91          final var constantW = iPoint.getHomW() / homW;
92          assertEquals(constantX, constantY, ABSOLUTE_ERROR);
93          assertEquals(constantY, constantW, ABSOLUTE_ERROR);
94          assertEquals(constantW, constantX, ABSOLUTE_ERROR);
95  
96          iPoint.setInhomogeneousCoordinates(inhomX, inhomY);
97          assertEquals(inhomX, iPoint.getInhomX(), 0.0);
98          assertEquals(inhomY, iPoint.getInhomY(), 0.0);
99          assertEquals(inhomX, iPoint.getX(), 0.0);
100         assertEquals(inhomY, iPoint.getY(), 0.0);
101     }

103     @Test
104     void testToHomogeneous() {
105         final var randomizer = new UniformRandomizer();
106         final var array = new double[INHOM_COORDS];
107         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
108 
109         final var iPoint = new InhomogeneousPoint2D(array);
110 
111         final var hPoint = iPoint.toHomogeneous();
112 
113         // check that inhomogeneous coordinates are almost equal
114         assertEquals(hPoint.getInhomX(), iPoint.getInhomX(), ABSOLUTE_ERROR);
115         assertEquals(hPoint.getInhomY(), iPoint.getInhomY(), ABSOLUTE_ERROR);
116 
117         // check that homogeneous coordinates are up to scale
118         final var scaleX = hPoint.getHomX() / iPoint.getHomX();
119         final var scaleY = hPoint.getHomY() / iPoint.getHomY();
120         final var scaleW = hPoint.getHomW() / iPoint.getHomW();
121 
122         assertEquals(scaleX, scaleY, ABSOLUTE_ERROR);
123         assertEquals(scaleY, scaleW, ABSOLUTE_ERROR);
124         assertEquals(scaleW, scaleX, ABSOLUTE_ERROR);
125     }

127     @Test
128     void testSetCoordinates() {
129         final var randomizer = new UniformRandomizer();
130         var array = new double[INHOM_COORDS];
131         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
132 
133         final var iPoint = new InhomogeneousPoint2D();
134         iPoint.setCoordinates(array);
135         var array2 = iPoint.asArray();
136         assertArrayEquals(array, array2, 0.0);
137 
138         // Force IllegalArgumentException
139         final var array3 = new double[INHOM_COORDS + 1];
140         final var iPoint3 = new InhomogeneousPoint2D();
141         assertThrows(IllegalArgumentException.class, () -> iPoint3.setCoordinates(array3));
142 
143         final var array4 = new double[INHOM_COORDS - 1];
144         final var iPoint4 = new InhomogeneousPoint2D();
145         assertThrows(IllegalArgumentException.class, () -> iPoint4.setCoordinates(array4));
146 
147         final var x = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
148         final var y = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
149         final var iPoint5 = new InhomogeneousPoint2D();
150         iPoint5.setCoordinates(x, y);
151         assertEquals(x, iPoint5.getX(), 0.0);
152         assertEquals(y, iPoint5.getY(), 0.0);
153 
154         final var inhomX = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
155         final var inhomY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
156         final var iPoint6 = new InhomogeneousPoint2D();
157         iPoint6.setInhomogeneousCoordinates(inhomX, inhomY);
158         assertEquals(inhomX, iPoint6.getInhomX(), 0.0);
159         assertEquals(inhomY, iPoint6.getInhomY(), 0.0);
160         assertEquals(inhomX, iPoint6.getX(), 0.0);
161         assertEquals(inhomY, iPoint6.getY(), 0.0);
162 
163         array = new double[INHOM_COORDS];
164         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
165         final var iPoint2 = new InhomogeneousPoint2D(array);
166         final var iPoint7 = new InhomogeneousPoint2D();
167         // pass another point to set coordinates
168         iPoint7.setCoordinates(iPoint2);
169         array2 = iPoint7.asArray();
170         assertArrayEquals(array, array2, 0.0);
171 
172         array = new double[HOM_COORDS];
173         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
174         final var hPoint = new HomogeneousPoint2D(array);
175         final var iPoint8 = new InhomogeneousPoint2D();
176         iPoint8.setCoordinates(hPoint);
177         array2 = iPoint8.asArray();
178         assertEquals(HOM_COORDS, array.length);
179         assertEquals(INHOM_COORDS, array2.length);
180         assertEquals(array[0] / array[2], array2[0], 0.0);
181         assertEquals(array[1] / array[2], array2[1], 0.0);
182     }

199     @Test
200     void testEqualsAndHashCode() {
201         final var array = new double[INHOM_COORDS];
202         final var randomizer = new UniformRandomizer();
203         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
204         var iPoint1 = new InhomogeneousPoint2D(array);
205         var iPoint2 = new InhomogeneousPoint2D(array);
206         assertTrue(iPoint1.equals(iPoint2, 0.0));
207         assertTrue(iPoint1.equals((Point2D) iPoint2, 0.0));
208         assertEquals(iPoint1.hashCode(), iPoint2.hashCode());
209 
210         array[0] = iPoint1.getX() + randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
211         array[1] = iPoint1.getY();
212         iPoint2 = new InhomogeneousPoint2D(array);
213         assertFalse(iPoint1.equals(iPoint2, 0.0));
214         assertFalse(iPoint1.equals((Point2D) iPoint2, 0.0));
215 
216         array[0] = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
217         array[1] = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
218         iPoint1 = new InhomogeneousPoint2D(array);
219         array[0] += 1.0;
220         iPoint2 = new InhomogeneousPoint2D(array);
221         assertTrue(iPoint1.equals(iPoint2, 1.0 + ABSOLUTE_ERROR));
222         assertTrue(iPoint1.equals((Point2D) iPoint2, 1.0 + ABSOLUTE_ERROR));
223         assertFalse(iPoint1.equals(iPoint2, 0.5));
224         assertFalse(iPoint1.equals(iPoint2, 0.5));
225 
226         // Testing equals from one homogeneous point
227         final var hArray = new double[HOM_COORDS];
228         randomizer.fill(hArray, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
229         array[0] = hArray[0] / hArray[2];
230         array[1] = hArray[1] / hArray[2];
231 
232         final var iPoint = new InhomogeneousPoint2D(array);
233         var hPoint = new HomogeneousPoint2D(hArray);
234         assertTrue(iPoint.equals(hPoint, ABSOLUTE_ERROR));
235         assertTrue(iPoint.equals((Point2D) hPoint, ABSOLUTE_ERROR));
236 
237         hArray[0] = iPoint.getHomX() + 1.0;
238         hArray[1] = iPoint.getHomY() + 1.0;
239         hArray[2] = iPoint.getHomW() + 1.0;
240         hPoint = new HomogeneousPoint2D(hArray);
241         assertFalse(iPoint.equals(hPoint, 0.0));
242         assertFalse(iPoint.equals((Point2D) hPoint, 0.0));
243 
244         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
245         final var iPoint3 = new InhomogeneousPoint2D(array);
246 
247         hArray[0] = iPoint3.getHomX() + iPoint3.getHomW();
248         hArray[1] = iPoint3.getHomY() + iPoint3.getHomW();
249         hArray[2] = iPoint3.getHomW();
250         final var hPoint2 = new HomogeneousPoint2D(hArray);
251         assertTrue(iPoint3.equals(hPoint2, 1.1));
252         assertTrue(iPoint3.equals((Point2D) hPoint2, 1.1));
253         assertFalse(iPoint3.equals(hPoint, 0.5));
254         assertFalse(iPoint3.equals((Point2D) hPoint2, 0.5));
255 
256         // Force IllegalArgumentException
257         assertThrows(IllegalArgumentException.class, () -> iPoint.equals(hPoint2, -ABSOLUTE_ERROR));
258         assertThrows(IllegalArgumentException.class, () -> iPoint.equals(iPoint3, -ABSOLUTE_ERROR));
259         assertThrows(IllegalArgumentException.class, () -> iPoint.equals((Point2D) iPoint3, -ABSOLUTE_ERROR));
260 
261         hPoint = new HomogeneousPoint2D();
262         final var iPoint4 = new InhomogeneousPoint2D();
263         assertTrue(iPoint4.equals(hPoint));
264         assertEquals(iPoint4.hashCode(), hPoint.hashCode());
265         final var iPoint5 = new InhomogeneousPoint2D(iPoint);
266         assertTrue(iPoint.equals(iPoint5));
267         assertEquals(iPoint.hashCode(), iPoint5.hashCode());
268         assertTrue(iPoint4.equals((Point2D) hPoint));
269         assertTrue(iPoint.equals((Point2D) iPoint5));
270     }

272     @Test
273     void testIsAtInfinity() {
274         final var array = new double[INHOM_COORDS];
275         final var randomizer = new UniformRandomizer();
276         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
277 
278         var iPoint = new InhomogeneousPoint2D(array);
279         iPoint.setX(Double.POSITIVE_INFINITY);
280         assertTrue(iPoint.isAtInfinity());
281 
282         iPoint = new InhomogeneousPoint2D(array);
283         iPoint.setY(Double.POSITIVE_INFINITY);
284         assertTrue(iPoint.isAtInfinity());
285 
286         iPoint = new InhomogeneousPoint2D(array);
287         assertFalse(iPoint.isAtInfinity());
288 
289         iPoint = new InhomogeneousPoint2D(array);
290         iPoint.setX(Double.NEGATIVE_INFINITY);
291         assertTrue(iPoint.isAtInfinity());
292 
293         iPoint = new InhomogeneousPoint2D(array);
294         iPoint.setY(Double.NEGATIVE_INFINITY);
295         assertTrue(iPoint.isAtInfinity());
296 
297         iPoint = new InhomogeneousPoint2D(array);
298         iPoint.setX(Double.NaN);
299         assertTrue(iPoint.isAtInfinity());
300 
301         iPoint = new InhomogeneousPoint2D(array);
302         iPoint.setY(Double.NaN);
303         assertTrue(iPoint.isAtInfinity());
304     }
