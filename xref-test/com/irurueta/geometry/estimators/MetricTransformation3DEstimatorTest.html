280     @Test
281     void testEstimateNoLMSE() throws NotReadyException, LockedException, CoincidentPointsException {
282 
283         var numValid = 0;
284         for (var t = 0; t < TIMES; t++) {
285 
286             final var randomizer = new UniformRandomizer();
287 
288             final var roll = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
289             final var pitch = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
290             final var yaw = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
291 
292             final var q = new Quaternion(roll, pitch, yaw);
293             q.normalize();
294 
295             final var translation = new double[3];
296             randomizer.fill(translation, MIN_TRANSLATION, MAX_TRANSLATION);
297 
298             final var scale = randomizer.nextDouble(MIN_SCALE, MAX_SCALE);
299 
300             final var transformation = new MetricTransformation3D(q, translation, scale);
301 
302             // generate random list of input points and transform them
303             final var inputPoints = new ArrayList<Point3D>();
304             for (var i = 0; i < MetricTransformation3DEstimator.MINIMUM_SIZE; i++) {
305                 final var x = randomizer.nextDouble(MIN_TRANSLATION, MAX_TRANSLATION);
306                 final var y = randomizer.nextDouble(MIN_TRANSLATION, MAX_TRANSLATION);
307                 final var z = randomizer.nextDouble(MIN_TRANSLATION, MAX_TRANSLATION);
308                 final var inputPoint = new InhomogeneousPoint3D(x, y, z);
309                 inputPoints.add(inputPoint);
310             }
311 
312             // transform points
313             final var outputPoints = transformation.transformPointsAndReturnNew(inputPoints);
314 
315             final var estimator = new MetricTransformation3DEstimator(this, inputPoints, outputPoints);
316 
317             reset();
318             assertEquals(0, estimateStart);
319             assertEquals(0, estimateEnd);
320             assertFalse(estimator.isLocked());
321 
322             final var transformation2 = estimator.estimate();
323 
324             assertEquals(1, estimateStart);
325             assertEquals(1, estimateEnd);
326             assertFalse(estimator.isLocked());
327 
328             reset();
329             assertEquals(0, estimateStart);
330             assertEquals(0, estimateEnd);
331             assertFalse(estimator.isLocked());
332 
333             final var transformation3 = new MetricTransformation3D();
334             estimator.estimate(transformation3);
335 
336             assertEquals(1, estimateStart);
337             assertEquals(1, estimateEnd);
338             assertFalse(estimator.isLocked());
339 
340             // check correctness of estimated transformations
341 
342             // transform points using transformation2
343             final var outputPoints2 = transformation2.transformPointsAndReturnNew(inputPoints);
344 
345             // check correctness
346             assertEquals(outputPoints.size(), outputPoints2.size());
347             var isValid = true;
348             for (var i = 0; i < outputPoints.size(); i++) {
349                 if (!outputPoints.get(i).equals(outputPoints2.get(i), ABSOLUTE_ERROR)) {
350                     isValid = false;
351                     break;
352                 }
353                 assertTrue(outputPoints.get(i).equals(outputPoints2.get(i), ABSOLUTE_ERROR));
354             }
355 
356             if (!isValid) {
357                 continue;
358             }
359 
360             final var q2 = transformation2.getRotation().toQuaternion();
361             q2.normalize();
362 
363             final var translation2 = transformation2.getTranslation();
364             final var scale2 = transformation2.getScale();
365 
366             assertEquals(q.getA(), q2.getA(), ABSOLUTE_ERROR);
367             assertEquals(q.getB(), q2.getB(), ABSOLUTE_ERROR);
368             assertEquals(q.getC(), q2.getC(), ABSOLUTE_ERROR);
369             assertEquals(q.getD(), q2.getD(), ABSOLUTE_ERROR);
370             assertArrayEquals(translation, translation2, ABSOLUTE_ERROR);
371             assertEquals(scale, scale2, ABSOLUTE_ERROR);
372 
373             // transform points using transformation3
374             final var outputPoints3 = transformation3.transformPointsAndReturnNew(inputPoints);
375 
376             // check correctness
377             assertEquals(outputPoints.size(), outputPoints2.size());
378             for (var i = 0; i < outputPoints.size(); i++) {
379                 assertTrue(outputPoints.get(i).equals(outputPoints3.get(i), ABSOLUTE_ERROR));
380             }
381 
382             final var q3 = transformation3.getRotation().toQuaternion();
383             q3.normalize();
384 
385             final var translation3 = transformation3.getTranslation();
386             final var scale3 = transformation3.getScale();
387 
388             assertEquals(q.getA(), q3.getA(), ABSOLUTE_ERROR);
389             assertEquals(q.getB(), q3.getB(), ABSOLUTE_ERROR);
390             assertEquals(q.getC(), q3.getC(), ABSOLUTE_ERROR);
391             assertEquals(q.getD(), q3.getD(), ABSOLUTE_ERROR);
392             assertArrayEquals(translation, translation3, ABSOLUTE_ERROR);
393             assertEquals(scale, scale3, ABSOLUTE_ERROR);
394 
395             numValid++;
396             break;
397         }
398 
399         assertTrue(numValid > 0);
400     }

524     @Test
525     void testEstimatePlanar() throws NotReadyException, LockedException, CoincidentPointsException {
526         var numValid = 0;
527         for (var t = 0; t < TIMES; t++) {
528             final var randomizer = new UniformRandomizer();
529 
530             final var roll = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
531             final var pitch = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
532             final var yaw = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
533 
534             final var q = new Quaternion(roll, pitch, yaw);
535             q.normalize();
536 
537             final var translation = new double[3];
538             randomizer.fill(translation, MIN_TRANSLATION, MAX_TRANSLATION);
539 
540             final var scale = randomizer.nextDouble(MIN_SCALE, MAX_SCALE);
541 
542             final var transformation = new MetricTransformation3D(q, translation, scale);
543 
544             // generate random list of input points and transform them
545             // generate random plane
546             final var a = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
547             final var b = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
548             final var c = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
549             final var d = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
550             final var plane = new Plane(a, b, c, d);
551 
552             final var inputPoints = new ArrayList<Point3D>();
553             for (var i = 0; i < MetricTransformation3DEstimator.WEAK_MINIMUM_SIZE; i++) {
554 
555                 final double homX;
556                 final double homY;
557                 final var homW = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
558                 final var homZ = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
559                 if (Math.abs(b) > ABSOLUTE_ERROR) {
560                     homX = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
561                     homY = -(a * homX + c * homZ + d * homW) / b;
562                 } else {
563                     homY = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
564                     homX = -(b * homY + c * homZ + d * homW) / a;
565                 }
566                 final var inputPoint = new HomogeneousPoint3D(homX, homY, homZ, homW);
567 
568                 assertTrue(plane.isLocus(inputPoint));
569 
570                 inputPoints.add(inputPoint);
571             }
572 
573             // transform points
574             final var outputPoints = transformation.transformPointsAndReturnNew(inputPoints);
575 
576             final var estimator = new MetricTransformation3DEstimator(this, inputPoints, outputPoints,
577                     true);
578 
579             reset();
580             assertEquals(0, estimateStart);
581             assertEquals(0, estimateEnd);
582             assertFalse(estimator.isLocked());
583 
584             final var transformation2 = estimator.estimate();
585 
586             assertEquals(1, estimateStart);
587             assertEquals(1, estimateEnd);
588             assertFalse(estimator.isLocked());
589 
590             reset();
591             assertEquals(0, estimateStart);
592             assertEquals(0, estimateEnd);
593             assertFalse(estimator.isLocked());
594 
595             final var transformation3 = new MetricTransformation3D();
596             estimator.estimate(transformation3);
597 
598             assertEquals(1, estimateStart);
599             assertEquals(1, estimateEnd);
600             assertFalse(estimator.isLocked());
601 
602             // check correctness of estimated transformations
603 
604             // transform points using transformation2
605             final var outputPoints2 = transformation2.transformPointsAndReturnNew(inputPoints);
606 
607             // check correctness
608             assertEquals(outputPoints.size(), outputPoints2.size());
609             var isValid = true;
610             for (var i = 0; i < outputPoints.size(); i++) {
611                 if (!outputPoints.get(i).equals(outputPoints2.get(i), ABSOLUTE_ERROR)) {
612                     isValid = false;
613                     break;
614                 }
615                 assertTrue(outputPoints.get(i).equals(outputPoints2.get(i), ABSOLUTE_ERROR));
616             }
617 
618             if (!isValid) {
619                 continue;
