30      private static final double MIN_ANGLE_DEGREES = 0.0;
31      private static final double MAX_ANGLE_DEGREES = 90.0;
32      private static final double ABSOLUTE_ERROR = 1e-6;
33  
34      private static final double MIN_RANDOM_VALUE = -10.0;
35      private static final double MAX_RANDOM_VALUE = 10.0;

37      private static final int TIMES = 50;

39      @Test
40      void testConstructor() throws AlgebraException, GeometryException {
41          for (var t = 0; t < TIMES; t++) {
42              // empty constructor
43              var accuracy = new Accuracy3D();
44  
45              // check default values
46              assertNull(accuracy.getCovarianceMatrix());
47              assertEquals(2.0, accuracy.getStandardDeviationFactor(), 0.0);
48              assertEquals(0.9544, accuracy.getConfidence(), 1e-2);
49              assertEquals(2.0 * NormalDist.cdf(2.0, 0.0, 1.0) - 1.0, accuracy.getConfidence(),
50                      0.0);
51              assertEquals(Double.POSITIVE_INFINITY, accuracy.getSmallestAccuracy(), 0.0);
52              assertEquals(Double.POSITIVE_INFINITY, accuracy.getLargestAccuracy(), 0.0);
53              assertEquals(Double.POSITIVE_INFINITY, accuracy.getAverageAccuracy(), 0.0);
54              assertEquals(3, accuracy.getNumberOfDimensions());
55  
56              // constructor with covariance matrix
57              final var randomizer = new UniformRandomizer();
58              final var semiAxesLengths = new double[Ellipsoid.DIMENSIONS];
59              var previous = 0.0;
60              for (var i = Ellipsoid.DIMENSIONS - 1; i >= 0; i--) {
61                  semiAxesLengths[i] = previous + randomizer.nextDouble();
62                  previous = semiAxesLengths[i];
63              }
64  
65              final var sqrSemiAxesLengths = new double[Ellipsoid.DIMENSIONS];
66              for (var i = 0; i < Ellipsoid.DIMENSIONS; i++) {
67                  sqrSemiAxesLengths[i] = semiAxesLengths[i] * semiAxesLengths[i];
68              }
69  
70              final var roll = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
71              final var pitch = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
72              final var yaw = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
73              final var rotation = new MatrixRotation3D(new Quaternion(roll, pitch, yaw));
74  
75              final var ellipsoid = new Ellipsoid(Point3D.create(), semiAxesLengths, rotation);
76  
77              final var rotationMatrix = rotation.asInhomogeneousMatrix();
78              final var covarianceMatrix = rotationMatrix.multiplyAndReturnNew(
79                      Matrix.diagonal(sqrSemiAxesLengths).multiplyAndReturnNew(rotationMatrix));
80  
81              accuracy = new Accuracy3D(covarianceMatrix);
82  
83              // check
84              assertSame(accuracy.getCovarianceMatrix(), covarianceMatrix);
85              assertEquals(2.0, accuracy.getStandardDeviationFactor(), 0.0);
86              assertEquals(0.9544, accuracy.getConfidence(), 1e-2);
87              assertEquals(2.0 * NormalDist.cdf(2.0, 0.0, 1.0) - 1.0, accuracy.getConfidence(),
88                      0.0);
89              assertEquals(accuracy.getStandardDeviationFactor() * semiAxesLengths[2],
90                      accuracy.getSmallestAccuracy(), ABSOLUTE_ERROR);
91              assertEquals(accuracy.getStandardDeviationFactor() * semiAxesLengths[0],
92                      accuracy.getLargestAccuracy(), ABSOLUTE_ERROR);
93              assertEquals(accuracy.getStandardDeviationFactor() * (semiAxesLengths[0] + semiAxesLengths[1]
94                              + semiAxesLengths[2]) / 3.0, accuracy.getAverageAccuracy(), ABSOLUTE_ERROR);
95              assertEquals(3, accuracy.getNumberOfDimensions());
96  
97              var ellipsoid2 = accuracy.toEllipsoid();
98              assertEquals(ellipsoid.getCenter(), ellipsoid2.getCenter());
99              assertArrayEquals(ArrayUtils.multiplyByScalarAndReturnNew(
100                     ellipsoid.getSemiAxesLengths(), accuracy.getStandardDeviationFactor()),
101                     ellipsoid2.getSemiAxesLengths(), ABSOLUTE_ERROR);
102             assertTrue(ellipsoid.getRotation().equals(ellipsoid2.getRotation(), ABSOLUTE_ERROR));
103             assertArrayEquals(ellipsoid2.getSemiAxesLengths(), ArrayUtils.multiplyByScalarAndReturnNew(semiAxesLengths,
104                             accuracy.getStandardDeviationFactor()), ABSOLUTE_ERROR);
105             assertTrue(ellipsoid2.getRotation().equals(rotation, ABSOLUTE_ERROR));
106 
107             // force IllegalArgumentException
108             final var wrong1 = new Matrix(1, 1);
109             assertThrows(IllegalArgumentException.class, () -> new Accuracy3D(wrong1));
110 
111             // force NonSymmetricPositiveDefiniteMatrixException
112             final var m = Matrix.diagonal(new double[]{
113                     Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY});
114             assertThrows(NonSymmetricPositiveDefiniteMatrixException.class, () -> new Accuracy3D(m));
115 
116             // constructor with confidence
117             final var conf = randomizer.nextDouble(0.0, 1.0);
118             accuracy = new Accuracy3D(conf);
119 
120             // check default values
121             assertNull(accuracy.getCovarianceMatrix());
122             assertTrue(accuracy.getStandardDeviationFactor() > 0.0);
123             assertEquals(NormalDist.invcdf((conf + 1.0) / 2.0, 0.0, 1.0),
124                     accuracy.getStandardDeviationFactor(), 0.0);
125             assertEquals(accuracy.getConfidence(), conf, 0.0);
126             assertEquals(Double.POSITIVE_INFINITY, accuracy.getSmallestAccuracy(), 0.0);
127             assertEquals(Double.POSITIVE_INFINITY, accuracy.getLargestAccuracy(), 0.0);
128             assertEquals(Double.POSITIVE_INFINITY, accuracy.getAverageAccuracy(), 0.0);
129             assertEquals(3, accuracy.getNumberOfDimensions());
130 
131             // force IllegalArgumentException
132             assertThrows(IllegalArgumentException.class, () -> new Accuracy3D(-1.0));
133             assertThrows(IllegalArgumentException.class, () -> new Accuracy3D(2.0));
134 
135             // constructor with covariance matrix and confidence
136             accuracy = new Accuracy3D(covarianceMatrix, conf);
137 
138             // check
139             assertSame(accuracy.getCovarianceMatrix(), covarianceMatrix);
140             assertTrue(accuracy.getStandardDeviationFactor() > 0.0);
141             assertEquals(NormalDist.invcdf((conf + 1.0) / 2.0, 0.0, 1.0),
142                     accuracy.getStandardDeviationFactor(), 0.0);
143             assertEquals(accuracy.getConfidence(), conf, 0.0);
144             assertEquals(accuracy.getStandardDeviationFactor() * semiAxesLengths[2],
145                     accuracy.getSmallestAccuracy(), ABSOLUTE_ERROR);
146             assertEquals(accuracy.getStandardDeviationFactor() * semiAxesLengths[0],
147                     accuracy.getLargestAccuracy(), ABSOLUTE_ERROR);
148             assertEquals(accuracy.getStandardDeviationFactor() * (semiAxesLengths[0] + semiAxesLengths[1]
149                             + semiAxesLengths[2]) / 3.0, accuracy.getAverageAccuracy(), ABSOLUTE_ERROR);
150             assertEquals(3, accuracy.getNumberOfDimensions());
151 
152             ellipsoid2 = accuracy.toEllipsoid();
153             assertEquals(ellipsoid.getCenter(), ellipsoid2.getCenter());
154             assertArrayEquals(ArrayUtils.multiplyByScalarAndReturnNew(ellipsoid.getSemiAxesLengths(),
155                             accuracy.getStandardDeviationFactor()), ellipsoid2.getSemiAxesLengths(), ABSOLUTE_ERROR);
156             assertTrue(ellipsoid.getRotation().equals(ellipsoid2.getRotation(), ABSOLUTE_ERROR));
157             assertArrayEquals(ArrayUtils.multiplyByScalarAndReturnNew(semiAxesLengths,
158                             accuracy.getStandardDeviationFactor()), ellipsoid2.getSemiAxesLengths(), ABSOLUTE_ERROR);
159             assertTrue(ellipsoid2.getRotation().equals(rotation, ABSOLUTE_ERROR));
160 
161             // force IllegalArgumentException
162             final var wrong = new Matrix(1, 1);
163             assertThrows(IllegalArgumentException.class, () -> new Accuracy3D(wrong, conf));
164             assertThrows(IllegalArgumentException.class, () -> new Accuracy3D(covarianceMatrix, -1.0));
165             assertThrows(IllegalArgumentException.class, () -> new Accuracy3D(covarianceMatrix, 2.0));
166 
167             // force NonSymmetricPositiveDefiniteMatrixException
168             assertThrows(NonSymmetricPositiveDefiniteMatrixException.class, () -> new Accuracy3D(m, conf));
169         }
170     }

241     @Test
242     void testFlattenTo2D() throws AlgebraException, GeometryException {
243         final var randomizer = new UniformRandomizer();
244         final var semiAxesLengths = new double[Ellipsoid.DIMENSIONS];
245         var previous = 0.0;
246         for (var i = Ellipsoid.DIMENSIONS - 1; i >= 0; i--) {
247             semiAxesLengths[i] = previous + randomizer.nextDouble();
248             previous = semiAxesLengths[i];
249         }
250 
251         final var sqrSemiAxesLengths = new double[Ellipsoid.DIMENSIONS];
252         for (var i = 0; i < Ellipsoid.DIMENSIONS; i++) {
253             sqrSemiAxesLengths[i] = semiAxesLengths[i] * semiAxesLengths[i];
254         }
255 
256         final var roll = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
257         final var pitch = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
258         final var yaw = Utils.convertToRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
259         final var rotation = new MatrixRotation3D(new Quaternion(roll, pitch, yaw));
260 
261         final var rotationMatrix = rotation.asInhomogeneousMatrix();
262         final var covarianceMatrix = rotationMatrix.multiplyAndReturnNew(Matrix.diagonal(sqrSemiAxesLengths)
263                 .multiplyAndReturnNew(rotationMatrix));
264 
265         final var accuracy = new Accuracy3D(covarianceMatrix);
266 
267         final var flattenedAccuracy = accuracy.flattenTo2D();
268 
269         final var ellipse = accuracy.intersectWithPlane();
270         final var flattenedEllipse = flattenedAccuracy.toEllipse();
271 
272         assertEquals(flattenedEllipse.getSemiMajorAxis(), ellipse.getSemiMajorAxis(), ABSOLUTE_ERROR);
273         assertEquals(flattenedEllipse.getSemiMinorAxis(), ellipse.getSemiMinorAxis(), ABSOLUTE_ERROR);
274         // because ellipses are symmetric, there is a rotation ambiguity
275         assertTrue(Math.abs(ellipse.getRotationAngle() - flattenedEllipse.getRotationAngle())
276                 <= ABSOLUTE_ERROR || Math.abs(Math.abs(ellipse.getRotationAngle()
277                 - flattenedEllipse.getRotationAngle()) - Math.PI) <= ABSOLUTE_ERROR);
278 
279         assertEquals(ellipse.getCenter(), Point2D.create());
280         assertEquals(flattenedEllipse.getCenter(), Point2D.create());
281     }

297                 .multiplyAndReturnNew(rotationMatrix));
