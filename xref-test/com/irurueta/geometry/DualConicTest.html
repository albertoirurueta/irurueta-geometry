529             l3 = line5.getC();

532             systemOfEquationsMatrix.setElementAt(4, 2, l2 * l2);
533             systemOfEquationsMatrix.setElementAt(4, 3, 2.0 * l1 * l3);
534             systemOfEquationsMatrix.setElementAt(4, 4, 2.0 * l2 * l3);
535             systemOfEquationsMatrix.setElementAt(4, 5, l3 * l3);
536         }

539                 systemOfEquationsMatrix);
540         decomposer.decompose();
541

543

545         double b = V.getElementAt(1, 5);
546         double c = V.getElementAt(2, 5);
547         double d = V.getElementAt(3, 5);
548         double e = V.getElementAt(4, 5);
549         double f = V.getElementAt(5, 5);
550             
551         Matrix dualConicLine = new Matrix(HOM_COORDS, 1);
552         dualConicLine.setElementAt(0, 0, line1.getA());
553         dualConicLine.setElementAt(1, 0, line1.getB());
554         dualConicLine.setElementAt(2, 0, line1.getC());
555         
556         double norm = com.irurueta.algebra.Utils.normF(dualConicLine);
557         dualConicLine.multiplyByScalar(1.0 / norm);            
558         
559         Matrix dualConicMatrix = new Matrix(DUAL_CONIC_ROWS, DUAL_CONIC_COLS);
560         dualConicMatrix.setElementAt(0, 0, a);
561         dualConicMatrix.setElementAt(0, 1, b);
562         dualConicMatrix.setElementAt(0, 2, d);
563         dualConicMatrix.setElementAt(1, 0, b);
564         dualConicMatrix.setElementAt(1, 1, c);
565         dualConicMatrix.setElementAt(1, 2, e);
566         dualConicMatrix.setElementAt(2, 0, d);
567         dualConicMatrix.setElementAt(2, 1, e);
568         dualConicMatrix.setElementAt(2, 2, f);
569         
570         norm = com.irurueta.algebra.Utils.normF(dualConicMatrix);
571         dualConicMatrix.multiplyByScalar(1.0 / norm);
572         
573         //find point where line is tangent to conic
574         Matrix homPointMatrix = dualConicMatrix.multiplyAndReturnNew(
575                 dualConicLine);
576         
577         //add director vector of tangent line to get a point outside of conic
578         double directVectorA = dualConicLine.getElementAtIndex(0);
579         double directVectorB = dualConicLine.getElementAtIndex(1);
580         double directVectorNorm = Math.sqrt(directVectorA * directVectorA +
581             directVectorB * directVectorB);
582         directVectorA /= directVectorNorm;
583         directVectorB /= directVectorNorm;
584         homPointMatrix.setElementAtIndex(0, homPointMatrix.getElementAtIndex(0) 
585                 + directVectorA * homPointMatrix.getElementAtIndex(2));
586         homPointMatrix.setElementAtIndex(1, homPointMatrix.getElementAtIndex(1) 
587                 + directVectorB * homPointMatrix.getElementAtIndex(2));
588         
589         norm = com.irurueta.algebra.Utils.normF(homPointMatrix);
590         homPointMatrix.multiplyByScalar(1.0 / norm);
591         
592         //get conic matrix by inversing dual conic matrix
593         Matrix conicMatrix = com.irurueta.algebra.Utils.inverse(
594                 dualConicMatrix);
595         
596         //find line vector outside of dual conic as the product of conic matrix
597         //and point outside of conic
598         Matrix outsideLineMatrix = conicMatrix.multiplyAndReturnNew(
599                 homPointMatrix);
600         
601         //instantiate line outside of dual conic using computed vector
602         Line2D outsideLine = new Line2D(outsideLineMatrix.toArray());
603         
604         //instantiate new dual conic instance
605         DualConic dualConic = new DualConic(dualConicMatrix);
606 
607         //check that initial 5 lines lie inside the dual conic
608         assertTrue(dualConic.isLocus(line1, LOCUS_THRESHOLD));
609         assertTrue(dualConic.isLocus(line2, LOCUS_THRESHOLD));
610         assertTrue(dualConic.isLocus(line3, LOCUS_THRESHOLD));
611         assertTrue(dualConic.isLocus(line4, LOCUS_THRESHOLD));
612         assertTrue(dualConic.isLocus(line5, LOCUS_THRESHOLD));
613         
614         //check line outside of dual conic
615         assertFalse(dualConic.isLocus(outsideLine, LOCUS_THRESHOLD));
616     }

619     public void testAngleBetweenLines() throws WrongSizeException, 
620             IllegalArgumentException, NonSymmetricMatrixException {
621         
622         //initial lines
623         Matrix lineMatrix1 = Matrix.createWithUniformRandomValues(HOM_COORDS, 1, 
624                 MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
625         Matrix lineMatrix2 = Matrix.createWithUniformRandomValues(HOM_COORDS, 1, 
626                 MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
627         
628         //transformation matrix
629         Matrix transform = Matrix.createWithUniformRandomValues(HOM_COORDS, 
630                 HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
631         
632         
633         //transform lines
634         Matrix tLine1 = transform.multiplyAndReturnNew(lineMatrix1);
635         Matrix tLine2 = transform.multiplyAndReturnNew(lineMatrix2);
636         
637         double norm1 = com.irurueta.algebra.Utils.normF(tLine1);
638         double norm2 = com.irurueta.algebra.Utils.normF(tLine2);
639         
640         double numerator = tLine1.transposeAndReturnNew().multiplyAndReturnNew(
641                 tLine2).getElementAt(0, 0);
642         
643         double cosAngle = numerator / (norm1 * norm2);
644         
645         double angle = Math.acos(cosAngle);
646         
647         //compute dual conic matrix as the product of transposed transform 
648         //matrix with itself
649         Matrix transposedTransform = transform.transposeAndReturnNew();
650         Matrix dualConicMatrix = transposedTransform.multiplyAndReturnNew(
651                 transform);
652         
653         //normalize conic matrix
654         double normDualConic = com.irurueta.algebra.Utils.normF(dualConicMatrix);
655         dualConicMatrix.multiplyByScalar(1.0 / normDualConic);
656         
657         DualConic dualConic = new DualConic(dualConicMatrix);
658         
659         Line2D line1 = new Line2D(lineMatrix1.getElementAt(0, 0), 
660                 lineMatrix1.getElementAt(1, 0), lineMatrix1.getElementAt(2, 0));
661         Line2D line2 = new Line2D(lineMatrix2.getElementAt(0, 0), 
662                 lineMatrix2.getElementAt(1, 0), lineMatrix2.getElementAt(2, 0));
663         
664         assertEquals(dualConic.angleBetweenLines(line1, line2), angle, 
665                 PRECISION_ERROR);

669     public void testArePerpendicularLines() throws WrongSizeException, 
670             DecomposerException, RankDeficientMatrixException,
671             IllegalArgumentException, NonSymmetricMatrixException {
672         
673         int numValid = 0;
674         for (int t = 0; t < TIMES; t++) {
675             //trying perpendicular angle
676             Matrix matrixLine1 = Matrix.createWithUniformRandomValues(HOM_COORDS, 1, 
677                     MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
678 
679             double norm = com.irurueta.algebra.Utils.normF(matrixLine1);
680             matrixLine1.multiplyByScalar(1.0 / norm);
681 
682             Matrix matrixLine2 = new Matrix(HOM_COORDS, 1);
683             matrixLine2.setElementAt(0, 0, matrixLine1.getElementAt(1, 0) + 
684                     matrixLine1.getElementAt(2, 0));
685             matrixLine2.setElementAt(1, 0, -matrixLine1.getElementAt(0, 0));
686             matrixLine2.setElementAt(2, 0, -matrixLine1.getElementAt(0, 0));
687 
688             norm = com.irurueta.algebra.Utils.normF(matrixLine2);
689             matrixLine2.multiplyByScalar(1.0 / norm);
690 
691             Matrix transform = Matrix.createWithUniformRandomValues(HOM_COORDS, 
692                     HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
693             while (com.irurueta.algebra.Utils.rank(transform) < 3) {
694                 transform = Matrix.createWithUniformRandomValues(HOM_COORDS, 
695                         HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
696             }
697 
698             Matrix invTransform = com.irurueta.algebra.Utils.inverse(transform);
699 
700             Matrix transformLineMatrix1 = transform.multiplyAndReturnNew(
701                     matrixLine1);
702             Matrix transformLineMatrix2 = transform.multiplyAndReturnNew(
703                     matrixLine2);
704 
705             Matrix transInvTransform = invTransform.transposeAndReturnNew();
706 
707             Matrix dualConicMatrix = transInvTransform.multiplyAndReturnNew(
708                     invTransform);
709             norm = com.irurueta.algebra.Utils.normF(dualConicMatrix);
710             dualConicMatrix.multiplyByScalar(1.0 / norm);
711 
712             Line2D transformLine1 = new Line2D(transformLineMatrix1.toArray());
713             Line2D transformLine2 = new Line2D(transformLineMatrix2.toArray());
714 
715             DualConic dualConic = new DualConic(dualConicMatrix);
716 
717             assertTrue(dualConic.arePerpendicularLines(transformLine1, 
718                     transformLine2, PERPENDICULAR_THRESHOLD));
719 
720             //trying non-perpendicular points
721             double dotProduct;
722 
723             matrixLine1 = Matrix.createWithUniformRandomValues(HOM_COORDS, 1, 
724                     MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
725             norm = com.irurueta.algebra.Utils.normF(matrixLine1);
726             matrixLine1.multiplyByScalar(1.0 / norm);
727 
728             matrixLine2 = Matrix.createWithUniformRandomValues(HOM_COORDS, 1, 
729                     MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
730             norm = com.irurueta.algebra.Utils.normF(matrixLine2);
731             matrixLine2.multiplyByScalar(1.0 / norm);
732 
733             dotProduct = matrixLine1.transposeAndReturnNew().multiplyAndReturnNew(
734                     matrixLine2).getElementAt(0, 0);
735 
736             //ensure lines are not perpendicular
737             while (Math.abs(dotProduct) < PERPENDICULAR_THRESHOLD) {
738                 matrixLine1 = Matrix.createWithUniformRandomValues(HOM_COORDS, 1, 
739                         MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
740                 norm = com.irurueta.algebra.Utils.normF(matrixLine1);
741                 matrixLine1.multiplyByScalar(1.0 / norm);
742 
743                 matrixLine2 = Matrix.createWithUniformRandomValues(HOM_COORDS, 1, 
744                         MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
745                 norm = com.irurueta.algebra.Utils.normF(matrixLine2);
746                 matrixLine2.multiplyByScalar(1.0 / norm);
747 
748                 dotProduct = matrixLine1.transposeAndReturnNew().
749                         multiplyAndReturnNew(matrixLine2).getElementAt(0, 0);
750             }
751 
752             transformLineMatrix1 = transform.multiplyAndReturnNew(matrixLine1);
753             transformLineMatrix2 = transform.multiplyAndReturnNew(matrixLine2);
754 
755             transformLine1 = new Line2D(transformLineMatrix1.toArray());
756             transformLine2 = new Line2D(transformLineMatrix2.toArray());
757 
758             if (dualConic.arePerpendicularLines(transformLine1, transformLine2,
759                     5.0 * PERPENDICULAR_THRESHOLD)) {
760                 continue;
761             }
762             assertFalse(dualConic.arePerpendicularLines(transformLine1,
763                     transformLine2, 5.0 * PERPENDICULAR_THRESHOLD));
764             numValid++;
765         }
766         
767         assertTrue(numValid > 0);
768     }
