47      @Test
48      void testConstructor() throws WrongSizeException, NotReadyException, LockedException, DecomposerException,
49              com.irurueta.algebra.NotAvailableException, ColinearPointsException, IllegalArgumentException,
50              ParallelVectorsException {
51  
52          final var randomizer = new UniformRandomizer();
53  
54          final var a = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
55          final var b = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
56          final var c = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
57          final var d = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
58  
59          var plane = new Plane();
60          assertEquals(0.0, plane.getA(), 0.0);
61          assertEquals(0.0, plane.getB(), 0.0);
62          assertEquals(0.0, plane.getC(), 0.0);
63          assertEquals(0.0, plane.getD(), 0.0);
64          assertFalse(plane.isNormalized());
65  
66          plane = new Plane(a, b, c, d);
67          assertEquals(a, plane.getA(), 0.0);
68          assertEquals(b, plane.getB(), 0.0);
69          assertEquals(c, plane.getC(), 0.0);
70          assertEquals(d, plane.getD(), 0.0);
71          assertFalse(plane.isNormalized());
72  
73          // instantiate plane using array
74          var array = new double[HOM_COORDS];
75          randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
76  
77          plane = new Plane(array);
78          assertEquals(array[0], plane.getA(), 0.0);
79          assertEquals(array[1], plane.getB(), 0.0);
80          assertEquals(array[2], plane.getC(), 0.0);
81  
82          // Force IllegalArgumentException
83          assertThrows(IllegalArgumentException.class, () -> new Plane(new double[HOM_COORDS + 1]));
84  
85          // find 1st point
86          final var array1 = new double[HOM_COORDS];
87          randomizer.fill(array1, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
88  
89          // find 2nd point
90          final var array2 = new double[HOM_COORDS];
91          randomizer.fill(array2, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
92  
93          final var lambda = randomizer.nextDouble();
94  
95          // find 3rd point as a co-linear point of 1st and 2nd point
96          final var array3 = new double[HOM_COORDS];
97          // array3 = lamda * array2
98          ArrayUtils.multiplyByScalar(array2, lambda, array3);
99          // array3 = array1 + array3 = array1 + lambda * array2
100         ArrayUtils.sum(array1, array3, array3);
101 
102         var point1 = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array1);
103         var point2 = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array2);
104         var point3 = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array3);
105 
106         // Force ColinearPointsException
107         assertTrue(Plane.areColinearPoints(point1, point2, point3));
108         final var finalPoint1 = point1;
109         final var finalPoint2 = point2;
110         final var finalPoint3 = point3;
111         assertThrows(ColinearPointsException.class, () -> new Plane(finalPoint1, finalPoint2, finalPoint3));
112 
113         // try with 3 non co-linear points
114         var m = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
115         final var decomposer = new SingularValueDecomposer(m);
116         decomposer.decompose();
117 
118         // ensure we create a matrix with 3 non-linear dependent rows
119         while (decomposer.getRank() < 3) {
120             m = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
121             decomposer.setInputMatrix(m);
122             decomposer.decompose();
123         }
124 
125         // V contains the right null-space of m in the last column, which is the
126         // plane where the points belong to in the rows of m
127         var v = decomposer.getV();
128 
129         point1 = new HomogeneousPoint3D(m.getElementAt(0, 0),
130                 m.getElementAt(0, 1),
131                 m.getElementAt(0, 2),
132                 m.getElementAt(0, 3));
133         point2 = new HomogeneousPoint3D(m.getElementAt(1, 0),
134                 m.getElementAt(1, 1),
135                 m.getElementAt(1, 2),
136                 m.getElementAt(1, 3));
137         point3 = new HomogeneousPoint3D(m.getElementAt(2, 0),
138                 m.getElementAt(2, 1),
139                 m.getElementAt(2, 2),
140                 m.getElementAt(2, 3));
141 
142         point1.normalize();
143         point2.normalize();
144         point3.normalize();
145 
146         final var arrayPlane = new double[HOM_COORDS];
147         arrayPlane[0] = v.getElementAt(0, 3);
148         arrayPlane[1] = v.getElementAt(1, 3);
149         arrayPlane[2] = v.getElementAt(2, 3);
150         arrayPlane[3] = v.getElementAt(3, 3);
151 
152         assertFalse(Plane.areColinearPoints(point1, point2, point3));
153         plane = new Plane(point1, point2, point3);
154 
155         // check correctness of obtained plane
156         final var scaleA = plane.getA() / arrayPlane[0];
157         final var scaleB = plane.getB() / arrayPlane[1];
158         final var scaleC = plane.getC() / arrayPlane[2];
159         final var scaleD = plane.getD() / arrayPlane[3];
160 
161         assertEquals(scaleA, scaleB, PRECISION_ERROR);
162         assertEquals(scaleB, scaleC, PRECISION_ERROR);
163         assertEquals(scaleC, scaleD, PRECISION_ERROR);
164         assertEquals(scaleD, scaleA, PRECISION_ERROR);
165 
166         // create plane passing through a point and laying on provided direction
167         // vectors
168         array = new double[HOM_COORDS];
169         randomizer.fill(array, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
170         final var point = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array);
171 
172         final var direction1 = new double[INHOM_COORDS];
173         randomizer.fill(direction1, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
174 
175         final var direction2 = new double[INHOM_COORDS];
176         randomizer.fill(direction2, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
177 
178         // normalize to compare with greater accuracy
179         final var norm1 = com.irurueta.algebra.Utils.normF(direction1);
180         final var direction1b = ArrayUtils.multiplyByScalarAndReturnNew(direction1, 1.0 / norm1);
181         final var norm2 = com.irurueta.algebra.Utils.normF(direction2);
182         final var direction2b = ArrayUtils.multiplyByScalarAndReturnNew(direction2, 1.0 / norm2);
183 
184         // director vector will already be normalized because direction vectors are normalized
185         final var directorVector = com.irurueta.algebra.Utils.crossProduct(direction1b, direction2b);
186 
187         // create plane using point and directions
188         var plane1 = new Plane(point, direction1, direction2);
189         // and create plane using point and director vector
190         var plane2 = new Plane(point, directorVector);
191 
192         // check correctness of instantiated planes
193         assertTrue(plane1.isLocus(point));
194         assertTrue(plane2.isLocus(point));
195 
196         final var directorVector1 = plane1.getDirectorVector();
197         assertEquals(directorVector1[0], plane1.getA(), 0.0);
198         assertEquals(directorVector1[1], plane1.getB(), 0.0);
199         assertEquals(directorVector1[2], plane1.getC(), 0.0);
200         final var directorVector2 = plane2.getDirectorVector();
201         assertEquals(directorVector2[0], plane2.getA(), 0.0);
202         assertEquals(directorVector2[1], plane2.getB(), 0.0);
203         assertEquals(directorVector2[2], plane2.getC(), 0.0);
204 
205         // check that both director vectors are equal
206         assertEquals(directorVector1[0], directorVector2[0], PRECISION_ERROR);
207         assertEquals(directorVector1[1], directorVector2[1], PRECISION_ERROR);
208         assertEquals(directorVector1[2], directorVector2[2], PRECISION_ERROR);
209 
210         // and equal to the estimated director vector
211         final var scale1 = directorVector1[0] / directorVector[0];
212         final var scale2 = directorVector1[1] / directorVector[1];
213         final var scale3 = directorVector1[2] / directorVector[2];
214 
215         assertEquals(scale1, scale2, PRECISION_ERROR);
216         assertEquals(scale2, scale3, PRECISION_ERROR);
217         assertEquals(scale3, scale1, PRECISION_ERROR);
218 
219         // both planes are equal
220         assertEquals(plane1.getA(), plane2.getA(), PRECISION_ERROR);
221         assertEquals(plane1.getB(), plane2.getB(), PRECISION_ERROR);
222         assertEquals(plane1.getC(), plane2.getC(), PRECISION_ERROR);
223         assertEquals(plane1.getD(), plane2.getD(), PRECISION_ERROR);
224 
225         // Force ParallelVectorsException
226         assertThrows(ParallelVectorsException.class, () -> new Plane(point, direction1, direction1));
227 
228         // Force IllegalArgumentException
229         final var wrongArray = new double[HOM_COORDS];
230         assertThrows(IllegalArgumentException.class, () -> new Plane(point, wrongArray, direction2));
231         assertThrows(IllegalArgumentException.class, () -> new Plane(point, direction1, wrongArray));
232         assertThrows(IllegalArgumentException.class, () -> new Plane(point, wrongArray));
233     }

278     void testSetParametersFromThreePoints() throws WrongSizeException, NotReadyException, LockedException,
279             DecomposerException, com.irurueta.algebra.NotAvailableException, ColinearPointsException {
280 
281         var m = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
282 
283         final var decomposer = new SingularValueDecomposer(m);
284         decomposer.decompose();
285 
286         while (decomposer.getRank() < 2) {
287             m = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
288             decomposer.setInputMatrix(m);
289             decomposer.decompose();
290         }
291 
292         // the right null-space of m (last column of V) contains the plane where
293         // the 3 points belong to
294         final var v = decomposer.getV();
295 
296         Point3D point1 = new HomogeneousPoint3D(m.getElementAt(0, 0),
297                 m.getElementAt(0, 1),
298                 m.getElementAt(0, 2),
299                 m.getElementAt(0, 3));
300         Point3D point2 = new HomogeneousPoint3D(m.getElementAt(1, 0),
301                 m.getElementAt(1, 1),
302                 m.getElementAt(1, 2),
303                 m.getElementAt(1, 3));
304         Point3D point3 = new HomogeneousPoint3D(m.getElementAt(2, 0),
305                 m.getElementAt(2, 1),
306                 m.getElementAt(2, 2),
307                 m.getElementAt(2, 3));
308 
309         point1.normalize();
310         point2.normalize();
311         point3.normalize();
312 
313         final var planeArray = new double[HOM_COORDS];
314         planeArray[0] = v.getElementAt(0, 3);
315         planeArray[1] = v.getElementAt(1, 3);
316         planeArray[2] = v.getElementAt(2, 3);
317         planeArray[3] = v.getElementAt(3, 3);
318 
319         var plane = new Plane();
320         assertFalse(Plane.areColinearPoints(point1, point2, point3));
321         plane.setParametersFromThreePoints(point1, point2, point3);
322 
323         // compare plane parameters respect to computed plane array parameters
324         // They will be equal up to scale
325         final var scaleA = plane.getA() / planeArray[0];
326         final var scaleB = plane.getB() / planeArray[1];
327         final var scaleC = plane.getC() / planeArray[2];
328         final var scaleD = plane.getD() / planeArray[3];
329 
330         assertEquals(scaleA, scaleB, PRECISION_ERROR);
331         assertEquals(scaleB, scaleC, PRECISION_ERROR);
332         assertEquals(scaleC, scaleD, PRECISION_ERROR);
333         assertEquals(scaleD, scaleA, PRECISION_ERROR);
334 
335         // all three points belong to the plane
336         // to increase accuracy
337         plane.normalize();
338         assertTrue(plane.isLocus(point1, PRECISION_ERROR));
339         assertTrue(plane.isLocus(point2, PRECISION_ERROR));
340         assertTrue(plane.isLocus(point3, PRECISION_ERROR));
341 
342         // Force ColinearPointsException
343         // find co-linear points using a third point being a linear combination
344         // of two other points so that a plane cannot be defined
345         final var randomizer = new UniformRandomizer();
346 
347         // find 1st point
348         final var array1 = new double[HOM_COORDS];
349         randomizer.fill(array1, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
350 
351         // find 2nd point
352         final var array2 = new double[HOM_COORDS];
353         randomizer.fill(array2, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
354 
355         final var lambda = randomizer.nextDouble();
356 
357         // find 3rd point as a co-linear point of 1st and 2nd point
358         final var array3 = new double[HOM_COORDS];
359         // array3 = lamda * array2
360         ArrayUtils.multiplyByScalar(array2, lambda, array3);
361         // array3 = array1 + array3 = array1 + lambda * array2
362         ArrayUtils.sum(array1, array3, array3);
363 
364         point1 = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array1);
365         point2 = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array2);
366         point3 = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array3);
367 
368         // Force ColinearPointsException
369         assertTrue(Plane.areColinearPoints(point1, point2, point3));
370         final var finalPoint1 = point1;
371         final var finalPoint2 = point2;
372         final var finalPoint3 = point3;
373         assertThrows(ColinearPointsException.class, () -> new Plane(finalPoint1, finalPoint2, finalPoint3));
374     }

489     @Test
490     void testIsLocus() throws WrongSizeException, NotReadyException, LockedException, DecomposerException,
491             com.irurueta.algebra.NotAvailableException {
492 
493         // randomly choose 3 points to find their corresponding plane
494         var m = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
495 
496         final var decomposer = new SingularValueDecomposer(m);
497         decomposer.decompose();
498 
499         while (decomposer.getRank() < 3) {
500             m = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
501             decomposer.setInputMatrix(m);
502             decomposer.decompose();
503         }
504 
505         final var v = decomposer.getV();
506 
507         final var plane = new Plane(v.getElementAt(0, 3),
508                 v.getElementAt(1, 3),
509                 v.getElementAt(2, 3),
510                 v.getElementAt(3, 3));
511 
512         final var point1 = new HomogeneousPoint3D(m.getElementAt(0, 0),
513                 m.getElementAt(0, 1),
514                 m.getElementAt(0, 2),
515                 m.getElementAt(0, 3));
516 
517         final var point2 = new HomogeneousPoint3D(m.getElementAt(1, 0),
518                 m.getElementAt(1, 1),
519                 m.getElementAt(1, 2),
520                 m.getElementAt(1, 3));
521 
522         final var point3 = new HomogeneousPoint3D(m.getElementAt(2, 0),
523                 m.getElementAt(2, 1),
524                 m.getElementAt(2, 2),
525                 m.getElementAt(2, 3));
526 
527         point1.normalize();
528         point2.normalize();
529         point3.normalize();
530 
531         // points belong to plane
532         assertTrue(plane.isLocus(point1, PRECISION_ERROR));
533         assertTrue(plane.isLocus(point2, PRECISION_ERROR));
534         assertTrue(plane.isLocus(point3, PRECISION_ERROR));
535 
536         // use plane director vector to find another point outside of plane
537         final var point4 = new HomogeneousPoint3D(point1);
538 
539         final var normDirectorVector = Math.sqrt(plane.getA() * plane.getA() + plane.getB() * plane.getB()

712     @Test
713     void testIntersection() throws WrongSizeException, NotReadyException, com.irurueta.algebra.NotAvailableException,
714             LockedException, DecomposerException, NoIntersectionException, ColinearPointsException {
715 
716         // Create random homogeneous coordinates for a point
717         final var m = Matrix.createWithUniformRandomValues(1, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
718 
719         // M is a 1x4 matrix having rank 1, hence its right null-space will have
720         // dimension 3. Each vector of the right null-space will follow equation:
721         // m * P = 0, hence each of those vectors will be a plane where the point
722         // will be locus, and hence the point will be the intersection of those
723         // 3 planes, which will be perpendicular among them
724         final var decomposer = new SingularValueDecomposer(m);
725         decomposer.decompose();
726 
727         final var v = decomposer.getV();
728 
729         final var point = new HomogeneousPoint3D(m.toArray());
730 
731         var plane1 = new Plane(v.getSubmatrixAsArray(0, 1,
732                 3, 1));
733         var plane2 = new Plane(v.getSubmatrixAsArray(0, 2,
734                 3, 2));
735         var plane3 = new Plane(v.getSubmatrixAsArray(0, 3,
736                 3, 3));
737 
738         assertTrue(plane1.getIntersection(plane2, plane3).equals(point, ABSOLUTE_ERROR));
739 
740         final var intersection = Point3D.create();
741         plane1.intersection(plane2, plane3, intersection);
742         assertTrue(intersection.equals(point, ABSOLUTE_ERROR));
743 
744         assertTrue(plane2.getIntersection(plane1, plane3).equals(point, ABSOLUTE_ERROR));
745 
746         plane2.intersection(plane1, plane3, intersection);
747         assertTrue(intersection.equals(point, ABSOLUTE_ERROR));
748 
749         assertTrue(plane3.getIntersection(plane1, plane2).equals(point, ABSOLUTE_ERROR));
750 
751         plane3.intersection(plane1, plane2, intersection);
752         assertTrue(intersection.equals(point, ABSOLUTE_ERROR));
753 
754         // Force NoIntersectionException by using two coincident or parallel planes
755         final var finalPlane1 = plane1;
756         final var finalPlane2 = plane2;
757         assertThrows(NoIntersectionException.class, () -> finalPlane1.getIntersection(finalPlane1, finalPlane2));
758         assertThrows(NoIntersectionException.class,
759                 () -> finalPlane1.intersection(finalPlane1, finalPlane2, intersection));
760 
761         // we could also find 7 random points to find 3 planes having one point
762         // in common, which will be their intersection
763         final var point1 = new HomogeneousPoint3D();
764         final var point2 = new HomogeneousPoint3D();
765         final var point3 = new HomogeneousPoint3D();
766         final var point4 = new HomogeneousPoint3D();

769         final var point7 = new HomogeneousPoint3D();
770

772         // defined as one row of the matrix)

774         var m2 = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
775         var m3 = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
776 
777         // set 1st point in common in all three matrices
778         m2.setSubmatrix(0, 0, 0, HOM_COORDS - 1,
779                 m1.getSubmatrix(0, 0, 0, HOM_COORDS - 1));
780         m3.setSubmatrix(0, 0, 0, HOM_COORDS - 1,
781                 m1.getSubmatrix(0, 0, 0, HOM_COORDS - 1));
782 
783         // ensure that all matrices have rank 3 (points are not co-linear)
784         while (com.irurueta.algebra.Utils.rank(m1) < 3 || com.irurueta.algebra.Utils.rank(m2) < 3
785                 || com.irurueta.algebra.Utils.rank(m3) < 3) {
786             // create random matrices again
787             m1 = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
788             m2 = Matrix.createWithUniformRandomValues(3, HOM_COORDS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);

790

792             m2.setSubmatrix(0, 0, 0, HOM_COORDS - 1,
793                     m1.getSubmatrix(0, 0, 0, HOM_COORDS - 1));
794             m3.setSubmatrix(0, 0, 0, HOM_COORDS - 1,
795                     m1.getSubmatrix(0, 0, 0, HOM_COORDS - 1));
796         }
797

799                 HOM_COORDS - 1));
800         point2.setCoordinates(m1.getSubmatrixAsArray(1, 0, 1,

813                 HOM_COORDS - 1));
814

819         plane3 = new Plane(point1, point6, point7);
