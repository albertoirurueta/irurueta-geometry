56      public void testCreate() {
57          Point3D point;
58          
59          point = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES);
60          assertEquals(point.getType(), CoordinatesType.HOMOGENEOUS_COORDINATES);
61          assertEquals(point.getDimensions(), 3);
62          assertEquals(point.getInhomX(), 0.0, 0.0);
63          assertEquals(point.getInhomY(), 0.0, 0.0);
64          assertEquals(point.getInhomZ(), 0.0, 0.0);
65          assertEquals(point.getHomX(), 0.0, 0.0);
66          assertEquals(point.getHomY(), 0.0, 0.0);
67          assertEquals(point.getHomZ(), 0.0, 0.0);
68          assertEquals(point.getHomW(), 1.0, 0.0);
69          
70          point = Point3D.create(CoordinatesType.INHOMOGENEOUS_COORDINATES);
71          assertEquals(point.getType(), 
72                  CoordinatesType.INHOMOGENEOUS_COORDINATES);
73          assertEquals(point.getDimensions(), 3);
74          assertEquals(point.getInhomX(), 0.0, 0.0);
75          assertEquals(point.getInhomY(), 0.0, 0.0);
76          assertEquals(point.getInhomZ(), 0.0, 0.0);
77          assertEquals(point.getHomX(), 0.0, 0.0);
78          assertEquals(point.getHomY(), 0.0, 0.0);
79          assertEquals(point.getHomZ(), 0.0, 0.0);
80          assertEquals(point.getHomW(), 1.0, 0.0);
81          
82          double[] array = new double[HOM_COORDS];
83          double[] iArray = new double[INHOM_COORDS];
84          
85          point = Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, array);
86          assertEquals(point.getType(), CoordinatesType.HOMOGENEOUS_COORDINATES);
87          
88          point = Point3D.create(CoordinatesType.INHOMOGENEOUS_COORDINATES, 
89                  iArray);
90          assertEquals(point.getType(), 
91                  CoordinatesType.INHOMOGENEOUS_COORDINATES);
92          
93          //Force IllegalArgumentException
94          try {
95              Point3D.create(CoordinatesType.INHOMOGENEOUS_COORDINATES, array);
96              fail("IllegalArgumentException expected but not thrown");
97          } catch (IllegalArgumentException ignore) { }
98          
99          try {
100             Point3D.create(CoordinatesType.HOMOGENEOUS_COORDINATES, iArray);
101             fail("IllegalArgumentException expected but not thrown");
102         } catch (IllegalArgumentException ignore) { }
103         
104         
105         point = Point3D.create(array);
106         assertEquals(point.getType(), Point3D.DEFAULT_COORDINATES_TYPE);
107         
108         //Force IllegalArgumentException
109         try {
110             Point3D.create(iArray);
111             fail("IllegalArgumentException expected but not thrown");
112         } catch (IllegalArgumentException ignore) { }
113         
114         point = Point3D.create();
115         assertEquals(point.getType(), Point3D.DEFAULT_COORDINATES_TYPE);        
116     }

291     @Test
292     public void testDistanceTo() {
293         
294         UniformRandomizer randomizer = new UniformRandomizer(new Random());
295         Point3D point1 = Point3D.create();
296         Point3D point2 = Point3D.create();
297         
298         point1.setInhomogeneousCoordinates(
299                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
300                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
301                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
302         point2.setInhomogeneousCoordinates(
303                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
304                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
305                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
306         
307         double diffX = point1.getInhomX() - point2.getInhomX();
308         double diffY = point1.getInhomY() - point2.getInhomY();
309         double diffZ = point1.getInhomZ() - point2.getInhomZ();
310         double distance = Math.sqrt(diffX * diffX + diffY * diffY + 
311                 diffZ * diffZ);
312         
313         //check distance
314         assertEquals(point1.distanceTo(point2), distance, ABSOLUTE_ERROR);
315         assertEquals(point2.distanceTo(point1), distance, ABSOLUTE_ERROR);
316         
317         //check distance to themselves
318         assertEquals(point1.distanceTo(point1), 0.0, ABSOLUTE_ERROR);
319         assertEquals(point2.distanceTo(point2), 0.0, ABSOLUTE_ERROR);
320     }

352     @Test
353     public void testDotProduct() {
354         UniformRandomizer randomizer = new UniformRandomizer(new Random());
355         Point3D point1 = Point3D.create();
356         Point3D point2 = Point3D.create();
357         Point3D point3 = Point3D.create();
358         
359         point1.setHomogeneousCoordinates(
360                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
361                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
362                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
363                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
364         point2.setHomogeneousCoordinates(-point1.getHomX(), -point1.getHomY(), 
365                 -point1.getHomZ(), -point1.getHomW());
366         point3.setHomogeneousCoordinates(
367                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
368                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
369                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
370                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
371         
372         //dot product for same point
373         assertEquals(point1.dotProduct(point1), 1.0, ABSOLUTE_ERROR);
374         //dot product for oposite signs
375         assertEquals(point1.dotProduct(point2), -1.0, ABSOLUTE_ERROR);
376         //dot product for random points
377         point1.normalize();
378         point3.normalize();
379         assertEquals(point1.dotProduct(point3), 
380                 point1.getHomX() * point3.getHomX() + 
381                 point1.getHomY() * point3.getHomY() + 
382                 point1.getHomZ() * point3.getHomZ() + 
383                 point1.getHomW() * point3.getHomW(), ABSOLUTE_ERROR);
384     }

386     @Test
387     public void testIsBetween() {
388         UniformRandomizer randomizer = new UniformRandomizer(new Random());
389         double valueBetween = randomizer.nextDouble(0.2, 0.8);
390         double valueOutside = 1.0 + valueBetween;
391         
392         Point3D point1 = Point3D.create();
393         Point3D point2 = Point3D.create();
394         
395         point1.setInhomogeneousCoordinates(
396                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
397                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
398                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
399         point2.setInhomogeneousCoordinates(
400                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE), 
401                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
402                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
403         
404         double diffX = point2.getInhomX() - point1.getInhomX();
405         double diffY = point2.getInhomY() - point1.getInhomY();
406         double diffZ = point2.getInhomZ() - point1.getInhomZ();
407         double dist = Math.sqrt(diffX * diffX + diffY * diffY + diffZ * diffZ);
408         
409         Point3D betweenPoint = Point3D.create();
410         betweenPoint.setInhomogeneousCoordinates(
411                 point1.getInhomX() + valueBetween * diffX, 
412                 point1.getInhomY() + valueBetween * diffY,
413                 point1.getInhomZ() + valueBetween * diffZ);
414 
415         Point3D outsidePoint = Point3D.create();
416         outsidePoint.setInhomogeneousCoordinates(
417                 point1.getInhomX() + valueOutside * diffX, 
418                 point1.getInhomY() + valueOutside * diffY,
419                 point1.getInhomZ() + valueOutside * diffZ);
420         
421         assertTrue(betweenPoint.isBetween(point1, point2));
422         assertFalse(outsidePoint.isBetween(point1, point2));
423         
424         //the same is true for a small threshold
425         assertTrue(betweenPoint.isBetween(point1, point2, ABSOLUTE_ERROR));
426         assertFalse(outsidePoint.isBetween(point1, point2, ABSOLUTE_ERROR));
427         
428         //with a large enough threshold, even outside point is considered to lie
429         //in between
430         assertTrue(betweenPoint.isBetween(point1, point2, 2.0 * dist));
431         assertTrue(outsidePoint.isBetween(point1, point2, 2.0 * dist));
432         
433         //Force IllegalArgumentException
434         try {
435             betweenPoint.isBetween(point1, point2, -ABSOLUTE_ERROR);
436             fail("IllegalArgumentException expected but not thrown");
437         } catch (IllegalArgumentException ignore) { }
438     }

440     @Test
441     public void testCentroid() {
442         UniformRandomizer randomizer = new UniformRandomizer(new Random());
443         int numPoints = randomizer.nextInt(MIN_POINTS, MAX_POINTS);
444         
445         List<Point3D> points = new ArrayList<>();
446         double x, y, z, meanX = 0.0, meanY = 0.0, meanZ = 0.0;
447         for (int i = 0; i < numPoints; i++) {
448             x = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
449             y = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
450             z = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
451             
452             points.add(new InhomogeneousPoint3D(x, y, z));
453             
454             meanX += x;
455             meanY += y;
456             meanZ += z;
457         }
458         
459         meanX /= numPoints;
460         meanY /= numPoints;
461         meanZ /= numPoints;
462         
463         Point3D mean1 = Point3D.create();
464         Point3D.centroid(points, mean1);
465         Point3D mean2 = Point3D.centroid(points);
466         
467         assertEquals(mean1, mean2);
468         assertEquals(mean1.getInhomX(), meanX, ABSOLUTE_ERROR);
469         assertEquals(mean1.getInhomY(), meanY, ABSOLUTE_ERROR);
470         assertEquals(mean1.getInhomZ(), meanZ, ABSOLUTE_ERROR);
471     }
