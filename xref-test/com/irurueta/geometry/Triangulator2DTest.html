16  package com.irurueta.geometry;
17  
18  import com.irurueta.statistics.UniformRandomizer;
19  import org.junit.jupiter.api.Test;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import static org.junit.jupiter.api.Assertions.*;
25  
26  class Triangulator2DTest {
27  
28      private static final int MIN_SIDES = 3;
29      private static final int MAX_SIDES = 12;
30  
31      private static final double MIN_RADIUS = 1.0;
32      private static final double MAX_RADIUS = 10.0;
33  
34      private static final double MIN_ANGLE_DEGREES = -180.0;
35      private static final double MAX_ANGLE_DEGREES = 180.0;
36  
37      private static final double ABSOLUTE_ERROR = 1e-8;
38  
39      @Test
40      void testConstants() {
41          assertEquals(3, Triangulator2D.MIN_VERTICES);
42          assertEquals(TriangulatorMethod.VAN_GOGH_TRIANGULATOR, Triangulator2D.DEFAULT_TRIANGULATOR_METHOD);
43      }
44  
45      @Test
46      void testCreateAndGetMethod() {
47          var triangulator = Triangulator2D.create();
48          assertNotNull(triangulator);
49  
50          // check method correctness
51          assertEquals(Triangulator2D.DEFAULT_TRIANGULATOR_METHOD, triangulator.getMethod());
52  
53          // create with method
54          triangulator = Triangulator2D.create(TriangulatorMethod.VAN_GOGH_TRIANGULATOR);
55  
56          // check method correctness
57          assertEquals(TriangulatorMethod.VAN_GOGH_TRIANGULATOR, triangulator.getMethod());
58      }
59  
60      @Test
61      void testTriangulate() throws NotEnoughVerticesException, TriangulatorException {
62  
63          final var randomizer = new UniformRandomizer();
64          final var sides = randomizer.nextInt(MIN_SIDES, MAX_SIDES);
65          final var radius = randomizer.nextDouble(MIN_RADIUS, MAX_RADIUS);
66  
67          // build vertices list
68          final var vertices = buildPolygonVertices(sides, radius);
69  
70          // build polygon
71          final var polygon = new Polygon2D(vertices);
72  
73          final var theta = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
74          // create point inside of vertex
75          final var point1 = new InhomogeneousPoint2D(
76                  radius / 2.0 * Math.cos(theta),
77                  radius / 2.0 * Math.sin(theta));
78  
79          // create point outside of vertex
80          final var point2 = new InhomogeneousPoint2D(
81                  2.0 * radius * Math.cos(theta),
82                  2.0 * radius * Math.sin(theta));
83  
84          final var triangulator = Triangulator2D.create();
85          final var triangles1 = triangulator.triangulate(polygon);
86          final var triangles2 = triangulator.triangulate(vertices);
87  
88          var signedArea1 = 0.0;
89          var area1 = 0.0;
90          var inside1 = false;
91          var inside2 = false;
92          for (final var triangle : triangles1) {
93              signedArea1 += triangle.getSignedArea();
94              area1 += triangle.getArea();
95              if (triangle.isInside(point1)) {
96                  inside1 = true;
97              }
98              if (triangle.isInside(point2)) {
99                  inside2 = true;
100             }
101         }
102 
103         assertEquals(signedArea1, polygon.getSignedArea(), ABSOLUTE_ERROR);
104         assertEquals(area1, polygon.getArea(), ABSOLUTE_ERROR);
105 
106         // check that point1 lies inside, but point2 does not
107         assertTrue(polygon.isInside(point1));
108         assertTrue(inside1);
109 
110         assertFalse(polygon.isInside(point2));
111         assertFalse(inside2);
112 
113         var signedArea2 = 0.0;
114         var area2 = 0.0;
115         inside1 = false;
116         for (final var triangle : triangles2) {
117             signedArea2 += triangle.getSignedArea();
118             area2 += triangle.getArea();
119             if (triangle.isInside(point1)) {
120                 inside1 = true;
121             }
122             if (triangle.isInside(point2)) {
123                 inside2 = true;
124             }
125         }
126 
127         assertEquals(signedArea2, polygon.getSignedArea(), ABSOLUTE_ERROR);
128         assertEquals(area2, polygon.getArea(), ABSOLUTE_ERROR);
129 
130         // check that point1 lies inside, but point2 does not
131         assertTrue(polygon.isInside(point1));
132         assertTrue(inside1);
133 
134         assertFalse(polygon.isInside(point2));
135         assertFalse(inside2);
136     }
137 
138     private static List<Point2D> buildPolygonVertices(final int sides, final double radius) {
139         final var vertices = new ArrayList<Point2D>(sides);
140         for (var i = 0; i < sides; i++) {
141             final var angle = (double) i / (double) sides * 2.0 * Math.PI;
142             final var vertex = new InhomogeneousPoint2D(radius * Math.cos(angle), radius * Math.sin(angle));
143             vertices.add(vertex);
144         }
145         return vertices;
146     }
147 }
