53      @Test
54      public void testConstructor() throws WrongSizeException, NotReadyException, 
55              LockedException, DecomposerException, NotAvailableException,
56              CoincidentPlanesException, NoIntersectionException,
57              CoincidentPointsException {
58          
59          //Create random homogeneous coordinates for a point
60          Matrix m = Matrix.createWithUniformRandomValues(1, HOM_COORDS, 
61                  MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
62          
63          //M is a 1x4 matrix having rank 1, hence its right null-space will have
64          //dimension 3. Each vector of the right null-space will follow equation:
65          //m * P = 0, hence each of those vectors will be a plane where the point
66          //will be locus, and hence the point will be the intersection of those
67          //3 planes, which will be perpendicular among them
68          SingularValueDecomposer decomposer = new SingularValueDecomposer(m);
69          decomposer.decompose();
70          
71          Matrix V = decomposer.getV();
72          
73          HomogeneousPoint3D point = new HomogeneousPoint3D(m.toArray());
74          
75          Plane plane1 = new Plane(V.getSubmatrixAsArray(0, 1,
76                  3, 1));
77          Plane plane2 = new Plane(V.getSubmatrixAsArray(0, 2,
78                  3, 2));
79          Plane plane3 = new Plane(V.getSubmatrixAsArray(0, 3,
80                  3, 3));
81          
82          //ensure that point is the intersection of all 3 points
83          assertTrue(plane1.getIntersection(plane2, plane3).equals(
84                  point.toInhomogeneous(), ABSOLUTE_ERROR));
85          
86          //test constructor from 2 planes
87          Line3D line = new Line3D(plane1, plane2);
88          
89          assertEquals(line.getPlane1(), plane1);
90          assertEquals(line.getPlane2(), plane2);
91          
92          assertTrue(line.isLocus(point));
93          
94          //Force ParallelPlanesException (by using the same plane)
95          line = null;
96          try {
97              line = new Line3D(plane1, plane1);
98              fail("CoincidentPlanesException expected but not thrown");
99          } catch (CoincidentPlanesException ignore) { }
100         assertNull(line);
101         
102         
103         //test constructor from 2 non-coincident points
104         m = Matrix.createWithUniformRandomValues(2, HOM_COORDS, 
105                 MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
106         while (Utils.rank(m) < 2) {
107             m = Matrix.createWithUniformRandomValues(2, HOM_COORDS, 
108                 MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
109         }
110         
111         Point3D point1 = new HomogeneousPoint3D(m.getSubmatrixAsArray(0,
112                 0, 0, HOM_COORDS - 1));
113         Point3D point2 = new HomogeneousPoint3D(m.getSubmatrixAsArray(1,
114                 0, 1, HOM_COORDS - 1));
115         
116         line = new Line3D(point1, point2);
117         
118         //ensure that both point1 and point2 belong to line 3D
119         assertTrue(line.isLocus(point1));
120         assertTrue(line.isLocus(point2));
121         
122         //planes get defined but we don't know their value, just that point1
123         //and point2 belong to those planes too
124         assertNotNull(line.getPlane1());
125         assertNotNull(line.getPlane2());
126         assertTrue(line.getPlane1().isLocus(point1));
127         assertTrue(line.getPlane1().isLocus(point2));
128         assertTrue(line.getPlane2().isLocus(point1));
129         assertTrue(line.getPlane2().isLocus(point2));
130         
131         
132         //Force CoincidentPointsException (by using the same point)
133         line = null;
134         try {
135             line = new Line3D(point1, point1);
136             fail("CoincidentPointsException expected but not thrown");
137         } catch (CoincidentPointsException ignore) { }
138         assertNull(line);
139     }

153         //will be locus, and hence the point will be the intersection of those

157

159

178         //but using the same plane...

180         assertTrue(Line3D.areCoincidentPlanes(plane2, plane2));
181         assertTrue(Line3D.areCoincidentPlanes(plane3, plane3));

287     @Test
288     public void testIsLocusDistanceAndClosestPoint() throws WrongSizeException, 
289             NotReadyException, LockedException, DecomposerException,
290             NotAvailableException, NoIntersectionException,
291             CoincidentPlanesException {
292         
293         //Create random homogeneous coordinates for a point
294         Matrix m = Matrix.createWithUniformRandomValues(1, HOM_COORDS, 
295                 MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
296         
297         //M is a 1x4 matrix having rank 1, hence its right null-space will have
298         //dimension 3. Each vector of the right null-space will follow equation:
299         //m * P = 0, hence each of those vectors will be a plane where the point
300         //will be locus, and hence the point will be the intersection of those
301         //3 planes, which will be perpendicular among them
302         SingularValueDecomposer decomposer = new SingularValueDecomposer(m);
303         decomposer.decompose();
304         
305         Matrix V = decomposer.getV();
306         
307         HomogeneousPoint3D point = new HomogeneousPoint3D(m.toArray());
308         
309         Plane plane1 = new Plane(V.getSubmatrixAsArray(0, 1,
310                 3, 1));
311         Plane plane2 = new Plane(V.getSubmatrixAsArray(0, 2,
312                 3, 2));
313         Plane plane3 = new Plane(V.getSubmatrixAsArray(0, 3,
314                 3, 3));
315         
316         //ensure that point is the intersection of all 3 points
317         assertTrue(plane1.getIntersection(plane2, plane3).equals(
318                 point.toInhomogeneous(), ABSOLUTE_ERROR));
319         
320         //test constructor from 2 planes
321         Line3D line = new Line3D(plane1, plane2);
322 
323         //test point is locus of line
324         assertTrue(line.isLocus(point));
325         assertTrue(line.isLocus(point, ABSOLUTE_ERROR));
326         
327         //Force IllegalArgumentException
328         try {
329             line.isLocus(point, -ABSOLUTE_ERROR);
330             fail("IllegalArgumentException expected but not thrown");
331         } catch (IllegalArgumentException ignore) { }
332         
333         //build another point at a given distance in perpendicular direction of
334         //line.
335         m = new Matrix(1, INHOM_COORDS);
336         m.setSubmatrix(0, 0, 0,
337                 INHOM_COORDS - 1, line.getDirection());
338         
339         decomposer.setInputMatrix(m);
340         decomposer.decompose();
341         
342         V = decomposer.getV();
343         
344         //because m has rank 1, the null space of m has rank 2, which are the 
345         //last 2 columns of V, which are perpendicular to m
346         Matrix perpendicular = V.getSubmatrix(0, 2,
347                 2, 2); //pick last column of V
348         
349         //determine amount of distance
350         UniformRandomizer randomizer = new UniformRandomizer(new Random());
351         double dist = randomizer.nextDouble(ABSOLUTE_ERROR, MAX_RANDOM_VALUE);
352         
353         Point3D distPoint = Point3D.create();
354         distPoint.setInhomogeneousCoordinates(
355                 point.getInhomX() + dist * perpendicular.getElementAtIndex(0), 
356                 point.getInhomY() + dist * perpendicular.getElementAtIndex(1), 
357                 point.getInhomZ() + dist * perpendicular.getElementAtIndex(2));
358         
359         //check that distPoint is indeed at distance dist
360         assertEquals(line.getDistance(distPoint), dist, ABSOLUTE_ERROR);
361         //check that distPoint is no longer locus of line 3D
362         assertFalse(line.isLocus(distPoint));
363         //but increasing enough the threshold, it becomes locus
364         assertTrue(line.isLocus(distPoint, dist));
365         
366         //because distPoint has moved in perpendicular direction from point,
367         //the closest point will be point
368         assertTrue(line.getClosestPoint(distPoint).equals(
369                 point.toInhomogeneous(), 2.0 * ABSOLUTE_ERROR));
370         assertTrue(line.getClosestPoint(distPoint, ABSOLUTE_ERROR).equals(
371                 point.toInhomogeneous(), ABSOLUTE_ERROR));
372         
373         Point3D closestPoint = Point3D.create();
374         line.closestPoint(distPoint, closestPoint);
375         assertTrue(closestPoint.equals(point.toInhomogeneous(), 
376                 ABSOLUTE_ERROR));
377         line.closestPoint(distPoint, closestPoint, ABSOLUTE_ERROR);
378         assertTrue(closestPoint.equals(point.toInhomogeneous(), 
379                 ABSOLUTE_ERROR));
380         
381         //Force IllegalArgumentException
382         try {
383             line.getClosestPoint(distPoint, -dist);
384             fail("IllegalArgumentException expected but not thrown");
385         } catch (IllegalArgumentException ignore) { }
386         try {
387             line.closestPoint(distPoint, closestPoint, -dist);
388             fail("IllegalArgumentException expected but not thrown");
389         } catch (IllegalArgumentException ignore) { }
390     }

424     @Test
425     public void testGetDirection() throws WrongSizeException, 
426             DecomposerException, CoincidentPointsException {
427         
428         //test constructor from 2 non-coincident points
429         Matrix m = Matrix.createWithUniformRandomValues(2, HOM_COORDS, 
430                 MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
431         while (Utils.rank(m) < 2) {
432             m = Matrix.createWithUniformRandomValues(2, HOM_COORDS, 
433                 MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
434         }
435         
436         Point3D point1 = new HomogeneousPoint3D(m.getSubmatrixAsArray(0,
437                 0, 0, HOM_COORDS - 1));
438         Point3D point2 = new HomogeneousPoint3D(m.getSubmatrixAsArray(1,
439                 0, 1, HOM_COORDS - 1));
440         
441         Line3D line = new Line3D(point1, point2);        
442         
443         double[] expectedDirection = Utils.crossProduct(
444                 line.getPlane1().getDirectorVector(), 
445                 line.getPlane2().getDirectorVector());
446 
447         double[] direction = line.getDirection();
448         
449         //check that direction is equal up to scale
450         double norm1 = Utils.normF(expectedDirection);
451         double norm2 = Utils.normF(direction);
452         
453         ArrayUtils.multiplyByScalar(expectedDirection, 1.0 / norm1, 
454                 expectedDirection);
455         ArrayUtils.multiplyByScalar(direction, 1.0 / norm2, direction);
456         
457         double[] diff = ArrayUtils.subtractAndReturnNew(direction, 
458                 expectedDirection);
459         
460         double normDiff = Utils.normF(diff);
461         assertTrue(normDiff < LARGE_ABSOLUTE_ERROR);
462     }
