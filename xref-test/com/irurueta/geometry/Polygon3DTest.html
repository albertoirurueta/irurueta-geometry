61          final int sides = randomizer.nextInt(MIN_SIDES, MAX_SIDES);
62          final double radius = randomizer.nextDouble(MIN_RADIUS, MAX_RADIUS);
63          final double theta = randomizer.nextDouble(MIN_ANGLE_DEGREES,
64                  MAX_ANGLE_DEGREES);

66          // build vertices list
67          final List<Point3D> vertices = buildPolygonVertices(sides, radius, theta);

70          Polygon3D polygon = new Polygon3D(vertices);

74          assertEquals(polygon.getVertices(), vertices);
75          assertEquals(polygon.getTriangulatorMethod(),
76                  Polygon3D.DEFAULT_TRIANGULATOR_METHOD);

79          final Iterator<Point3D> iterator2 = vertices.iterator();

82          Point3D vertex2;

84              vertex1 = iterator1.next();
85              vertex2 = iterator2.next();

175     @Test
176     public void testTriangulateIsTriangulatedAndArea()
177             throws NotEnoughVerticesException, TriangulatorException {
178         final UniformRandomizer randomizer = new UniformRandomizer(new Random());
179         final int sides = randomizer.nextInt(MIN_SIDES, MAX_SIDES);
180         final double radius = randomizer.nextDouble(MIN_RADIUS, MAX_RADIUS);
181         final double theta = randomizer.nextDouble(MIN_ANGLE_DEGREES,
182                 MAX_ANGLE_DEGREES);
183 
184         // build vertices list
185         final List<Point3D> vertices = buildPolygonVertices(sides, radius, theta);
186 
187         final Polygon3D polygon = new Polygon3D(vertices);
188 
189         final VanGoghTriangulator3D triangulator = new VanGoghTriangulator3D();
190 
191         // expected area is the sum of the areas of the triangles forming this
192         // polygon
193         double area = 0.0;
194         for (final Triangle3D triangle : triangulator.triangulate(vertices)) {
195             area += triangle.getArea();
196         }
197 
198         assertFalse(polygon.isTriangulated());
199         final List<Triangle3D> triangles = polygon.getTriangles();
200         assertTrue(polygon.isTriangulated());
201 
202         double areaTriangles = 0.0;
203         for (final Triangle3D triangle : triangles) {
204             areaTriangles += triangle.getArea();
205         }
206 
207         assertEquals(polygon.getArea(), area, ABSOLUTE_ERROR);
208         assertEquals(polygon.getArea(), areaTriangles, ABSOLUTE_ERROR);
209 
210         // triangulate again
211         polygon.triangulate();
212 
213         assertTrue(polygon.isTriangulated());
214     }

216     @Test
217     public void testPerimeter() throws NotEnoughVerticesException {
218         final UniformRandomizer randomizer = new UniformRandomizer(new Random());
219         int sides = randomizer.nextInt(MIN_SIDES, MAX_SIDES);
220         final double radius = randomizer.nextDouble(MIN_RADIUS, MAX_RADIUS);
221         final double theta = randomizer.nextDouble(MIN_ANGLE_DEGREES,
222                 MAX_ANGLE_DEGREES);
223 
224         // build vertices list
225         final List<Point3D> vertices = buildPolygonVertices(sides, radius, theta);
226 
227         final Polygon3D polygon = new Polygon3D(vertices);
228 
229         final Iterator<Point3D> iterator = vertices.iterator();
230         Point3D prevVertex = iterator.next();
231         Point3D curVertex;
232         double perimeter = 0.0;
233         while (iterator.hasNext()) {
234             curVertex = iterator.next();
235             perimeter += curVertex.distanceTo(prevVertex);
236             prevVertex = curVertex;
237         }
238 
239         // compare distance of last vertex with first one
240         perimeter += prevVertex.distanceTo(vertices.get(0));
241 
242         assertEquals(polygon.getPerimeter(), perimeter, ABSOLUTE_ERROR);
243 
244         // Test for a triangle
245         sides = 3;
246         final List<Point3D> vertices2 = buildPolygonVertices(sides, radius, theta);
247         final Triangle3D triangle = new Triangle3D(vertices2.get(0), vertices2.get(1),
248                 vertices2.get(2));
249         polygon.setVertices(vertices2);
250         assertEquals(polygon.getPerimeter(), triangle.getPerimeter(),
251                 ABSOLUTE_ERROR);
252     }

334     @Test
335     public void testIsLocusGetShortestDistanceAndClosestPoint()
336             throws NotEnoughVerticesException, TriangulatorException,
337             CoincidentPointsException {
338 
339         for (int t = 0; t < TIMES; t++) {
340             final UniformRandomizer randomizer = new UniformRandomizer(new Random());
341             final int sides = randomizer.nextInt(MIN_SIDES, MAX_SIDES);
342             final double radius = randomizer.nextDouble(MIN_RADIUS, MAX_RADIUS);
343             double dist = randomizer.nextDouble(2.0 * ABSOLUTE_ERROR, radius / 2.0);
344             final double phi = 0.0;
345 
346             final double theta = randomizer.nextDouble(0.0,
347                     2.0 * MAX_ANGLE_DEGREES) * Math.PI / 180.0;
348 
349             int n = (int) (theta / (2 * Math.PI) * (double) sides);
350 
351             final List<Point3D> vertices = buildPolygonVertices(sides, radius, phi);
352 
353             final Point3D point1 = vertices.get(n);
354             n = (n + 1) % sides;
355             final Point3D point2 = vertices.get(n);
356 
357             final Line3D line = new Line3D(point1, point2);
358             line.normalize();
359 
360             final Line2D line2D = new Line2D(
361                     new InhomogeneousPoint2D(point1.getInhomX(), point1.getInhomY()),
362                     new InhomogeneousPoint2D(point2.getInhomX(), point2.getInhomY()));
363             // to increase accuracy
364             line2D.normalize();
365             assertEquals(point1.getInhomZ(), point2.getInhomZ(), ABSOLUTE_ERROR);
366 
367             final double[] direction = line2D.getDirectorVector();
368             final double norm = Utils.normF(direction);
369             ArrayUtils.multiplyByScalar(direction, 1.0 / norm, direction);
370 
371             // find point laying on line between polygon vertices
372             final Point3D testPoint = new InhomogeneousPoint3D(
373                     radius * Math.cos(theta) * Math.cos(phi),
374                     radius * Math.sin(theta) * Math.cos(phi),
375                     radius * Math.sin(phi));
376             // point below is locus of polygon
377             final Point3D locusPoint = line.getClosestPoint(testPoint);
378             // locus point is between point1 and point2
379             assertTrue(locusPoint.isBetween(point1, point2));
380 
381             // move point a little bit away from locus
382             final Point3D notLocusPoint = new InhomogeneousPoint3D(
383                     locusPoint.getInhomX() + dist * direction[0],
384                     locusPoint.getInhomY() + dist * direction[1],
385                     locusPoint.getInhomZ());
386             // ensure that notLocusPoint lies outside of polygon
387             final Polygon3D polygon = new Polygon3D(vertices);
388 
389             if (polygon.isInside(notLocusPoint)) {
390                 // change sign of dist to move point in opposite direction
391                 dist *= -1.0;
392                 notLocusPoint.setInhomogeneousCoordinates(
393                         locusPoint.getInhomX() + dist * direction[0],
394                         locusPoint.getInhomY() + dist * direction[1],
395                         locusPoint.getInhomZ());
396             }
397             assertFalse(polygon.isInside(notLocusPoint));
398 
399             // check that notLocusPoint is at distance dist from line
400             assertEquals(line.getDistance(notLocusPoint), Math.abs(dist),
401                     ABSOLUTE_ERROR);
402 
403             assertTrue(polygon.isLocus(locusPoint));
404             assertTrue(polygon.isLocus(locusPoint, ABSOLUTE_ERROR));
405 
406             // because point is locus, the shortest distance is zero and it is the
407             // closest point
408             assertEquals(polygon.getShortestDistance(locusPoint), 0.0,
409                     ABSOLUTE_ERROR);
410             assertTrue(polygon.getClosestPoint(locusPoint).equals(locusPoint,
411                     ABSOLUTE_ERROR));
412             final Point3D closestPoint = Point3D.create();
413             polygon.closestPoint(locusPoint, closestPoint);
414             assertTrue(closestPoint.equals(locusPoint, ABSOLUTE_ERROR));
415             assertTrue(closestPoint.equals(locusPoint, ABSOLUTE_ERROR));
416 
417             assertFalse(polygon.isLocus(notLocusPoint));
418             assertFalse(polygon.isLocus(notLocusPoint, EPS));
419 
420             // not locus point is at distance dist from polygon
421             assertEquals(polygon.getShortestDistance(notLocusPoint), Math.abs(dist),
422                     ABSOLUTE_ERROR);
423 
424             // and the closest point to polygon is locusPoint
425             assertTrue(polygon.getClosestPoint(notLocusPoint).equals(
426                     new InhomogeneousPoint3D(locusPoint),
427                     ABSOLUTE_ERROR));
428 
429             polygon.closestPoint(notLocusPoint, closestPoint);
430             assertTrue(closestPoint.equals(new InhomogeneousPoint3D(locusPoint),
431                     ABSOLUTE_ERROR));
432             assertTrue(closestPoint.equals(new InhomogeneousPoint3D(locusPoint),
433                     ABSOLUTE_ERROR));
434 
435             // with a large enough threshold, not locus point is considered as locus
436             assertTrue(polygon.isLocus(notLocusPoint, radius * radius));
437 
438             // all vertices of polygon are also locus
439             for (final Point3D vertex : vertices) {
440                 assertTrue(polygon.isLocus(vertex));
441                 assertTrue(polygon.isLocus(vertex, ABSOLUTE_ERROR));
442                 // because vertices are locus, shortest distance is 0.0 and it is
443                 // a closest point
444                 assertEquals(polygon.getShortestDistance(vertex), 0.0,
445                         ABSOLUTE_ERROR);
446                 assertTrue(polygon.getClosestPoint(vertex).equals(vertex,
447                         ABSOLUTE_ERROR));
448                 polygon.closestPoint(vertex, closestPoint);
449                 assertTrue(closestPoint.equals(vertex, ABSOLUTE_ERROR));
450                 assertTrue(closestPoint.equals(vertex, ABSOLUTE_ERROR));
451             }
452 
453             // Force IllegalArgumentException
454             try {
455                 polygon.isLocus(locusPoint, -ABSOLUTE_ERROR);
456                 fail("IllegalArgumentException expected but not thrown");
457             } catch (final IllegalArgumentException ignore) {
458             }
459         }
460     }

715 
716         // X: -1, Y: 0, Z: 5 (W: 1)
717         final Point3D v1 = new InhomogeneousPoint3D(-1.0, 0.0, 5.0);
718         // X: 1, Y: 0, Z: 5 (W: 1)

722         // X: 0, Y: 1, Z: 5 (W: 1)
723         final Point3D v4 = new InhomogeneousPoint3D(0.0, 1.0, 5.0);
724 
725         // parallelepiped
726         final List<Point3D> vertices1 = new ArrayList<>();
727         final List<Point3D> vertices2 = new ArrayList<>();
728         vertices1.add(v1);
729         vertices1.add(v2);
730         vertices1.add(v3);
731         vertices1.add(v4);
732 
733         vertices2.add(v4);
734         vertices2.add(v3);
735         vertices2.add(v2);
736         vertices2.add(v1);
737 
738         final Polygon3D polygon1 = new Polygon3D(vertices1);
739         // because order of vertices is reversed, orientation will have opposite
740         // direction
741         final Polygon3D polygon2 = new Polygon3D(vertices2);
742 
743         double angle = Polygon3D.getAngleBetweenPolygons(vertices1, vertices2);

752                 ABSOLUTE_ERROR);
753         assertEquals(angle, Math.PI, ABSOLUTE_ERROR);

755         // trying with same polygon will return an angle of zero
756         angle = Polygon3D.getAngleBetweenPolygons(vertices1, vertices1);

758         angle = Polygon3D.getAngleBetweenPolygons(vertices1, vertices1,

760         assertEquals(angle, 0.0, ABSOLUTE_ERROR);

762         angle = Polygon3D.getAngleBetweenPolygons(polygon1, polygon1);
763         assertEquals(angle, 0.0, ABSOLUTE_ERROR);

765                 ABSOLUTE_ERROR);

773         } catch (final IllegalArgumentException ignore) {

779         } catch (final IllegalArgumentException ignore) {

819     private List<Point3D> buildPolygonVertices(
820             final int sides, final double radius,
821             final double theta) {
822         final List<Point3D> vertices = new ArrayList<>(sides);
823         Point3D vertex;
824         for (int i = 0; i < sides; i++) {
825             final double angle = (double) i / (double) sides * 2.0 * Math.PI;
826             vertex = new InhomogeneousPoint3D(
827                     radius * Math.cos(angle) * Math.cos(theta),
828                     radius * Math.sin(angle) * Math.cos(theta),
829                     radius * Math.sin(theta));
830             vertices.add(vertex);
831         }
832         return vertices;
833     }

835     private double[] absArray(final double[] array) {
836         final double[] out = new double[array.length];
837         for (int i = 0; i < array.length; i++) {
838             out[i] = Math.abs(array[i]);
839         }
840         return out;
841     }
