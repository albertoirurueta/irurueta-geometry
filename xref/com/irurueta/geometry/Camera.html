16  package com.irurueta.geometry;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  /**
22   * A camera defines relations between 3D and 2D worlds.
23   */
24  public abstract class Camera {
25      
26      /**
27       * Constant defining default camera type.
28       */
29      public static final CameraType DEFAULT_CAMERA_TYPE = 
30              CameraType.PINHOLE_CAMERA;
31      
32      /**
33       * Projects a 3D point into a 2D point in a retinal plane.
34       * @param point 3D point to be projected.
35       * @return 2D projected point.
36       */
37      public Point2D project(Point3D point) {
38          Point2D projected = Point2D.create(
39                  CoordinatesType.HOMOGENEOUS_COORDINATES);
40          project(point, projected);
41          return projected;
42      }
43      
44      /**
45       * Projects a 3D point into a 2D point in a retinal plane and stores the
46       * result into provided instance.
47       * @param inputPoint 3D point to be projected.
48       * @param result Instance where 2D projected point result is stored.
49       */
50      public abstract void project(Point3D inputPoint, Point2D result);
51      
52      /**
53       * Projects 3D points into 2D points in a retinal plane.
54       * @param points 3D points to be projected.
55       * @return 2D projected points.
56       */
57      public List<Point2D> project(List<Point3D> points){
58          List<Point2D> projected = new ArrayList<>(points.size());
59          project(points, projected);
60          return projected;
61      }
62      
63      /**
64       * Projects 3D points into 2D points in a retinal plane and stores the
65       * result into provided list.
66       * Note that if result list is not empty, it will be cleared when calling
67       * this method and then the estimated 2D points will be stored in it.
68       * @param inputPoints 3D points to be projected.
69       * @param result Instance where 2D projected points are stored.
70       */
71      public void project(List<Point3D> inputPoints, List<Point2D> result) {
72          result.clear();
73          for (Point3D point : inputPoints) {
74              result.add(project(point));
75          }
76      }
77      
78      /**
79       * Backprojects a 2D line into a 3D plane.
80       * @param line 2D line to be backprojected.
81       * @return 3D plane that has been backprojected.
82       */
83      public Plane backProject(Line2D line) {
84          Plane plane = new Plane();
85          backProject(line, plane);
86          return plane;
87      }
88  
89      //M^T * PLANE = 0
90      //m^T * l = 0
91      //m = P * M
92      //(P * M)^T * l = 0 --> M^T * (P^T * l) = 0
93      // PLANE = P^T * l
94      
95      /**
96       * Backprojects a line into a plane and stores the result into provided
97       * instance.
98       * @param line 2D line to be backprojected.
99       * @param result Instance where computed backprojected 3D plane data is 
100      * stored.
101      */
102     public abstract void backProject(Line2D line, Plane result);    
103     
104     /**
105      * Backprojects provided 2D lines into their corresponding 3D planes.
106      * @param lines 2D lines to be backprojected.
107      * @return 3D planes that have been backprojected.
108      */
109     public List<Plane> backProjectLines(List<Line2D> lines) {
110         List<Plane> planes = new ArrayList<>(lines.size());
111         backProjectLines(lines, planes);
112         return planes;
113     }
114     
115     /**
116      * Backprojects provided 2D lines into their corresponding 3D planes and
117      * stores the result into provided list.
118      * Note that if result list is not empty, its contents will be cleared when
119      * calling this method and then the estimated 3D planes will be stored in it.
120      * @param lines 2D lines to be backprojected.
121      * @param result 3D planes that have been backprojected.
122      */
123     public void backProjectLines(List<Line2D> lines, List<Plane> result) {
124         result.clear();
125         for (Line2D line : lines) {
126             result.add(backProject(line));
127         }
128     }
129 
130     //Note: multiple point3D points can be backprojected (indeed is a ray of 
131     //light). In other words, solution is not unique
132 
133     /**
134      * Backprojects provided 2D point into a 3D point.
135      * Notice that estimated solution is not unique, since backprojecting a 2D
136      * point results in an infinite number of 3D points located in the same
137      * ray of light.
138      * This method only returns one possible solution. Any possible solution can
139      * be computed as a linear combination between the camera center and the
140      * estimated point.
141      * @param point 2D point to be backprojected.
142      * @return A backprojected 3D point.
143      * @throws CameraException thrown if 2D point cannot be backprojected 
144      * because camera is degenerate.
145      */
146     public Point3D backProject(Point2D point) throws CameraException {
147         Point3D result = Point3D.create();
148         backProject(point, result);
149         return result;
150     }
151     
152     /**
153      * Backprojects provided 2D point into a 3D point and stores the result into
154      * provided instance.
155      * Notice that estimated solution is not unique, since backprojecting a 2D
156      * point results in an infinite number of 3D points located in the same
157      * ray of light.
158      * This method only computes one possible solution. Any other solutionc an
159      * be computed as a linear combination between the camera center and the
160      * estimated backprojeted point.
161      * @param point 2D point to be backprojected.
162      * @param result Instance where backprojected 3D point data will be stored
163      * @throws CameraException thrown if 2D point cannot be backprojected 
164      * because camera is degenerate.
165      */
166     public abstract void backProject(Point2D point, Point3D result) 
167             throws CameraException;
168     
169     /**
170      * Backprojects provided 2D points into their corresponding 3D points.
171      * @param points 2D points to be backprojected.
172      * @return 3D points that have been backprojected.
173      * @throws CameraException thrown if 2D point cannot be backprojected 
174      * because camera is degenerate.
175      */
176     public List<Point3D> backProjectPoints(List<Point2D> points) 
177             throws CameraException {
178         
179         List<Point3D> result = new ArrayList<>(points.size());
180         backProjectPoints(points, result);
181         return result;
182     }
183     
184     /**
185      * Backprojects provided 2D points into their corresponding 3D points and
186      * stores the result into provided list.
187      * Note that if result list is not empty, its contents will be cleared when
188      * calling this method and then the estimated 3D points will be stored in it.
189      * @param points 2D points to be backprojected.
190      * @param result 3D points that have been backprojected.
191      * @throws CameraException thrown if 2D point cannot be backprojected 
192      * because camera is degenerate.
193      */
194     @SuppressWarnings("WeakerAccess")
195     public void backProjectPoints(List<Point2D> points, List<Point3D> result)
196             throws CameraException {
197         
198         result.clear();
199         for (Point2D point : points) {
200             result.add(backProject(point));
201         }
202     }
203         
204     /**
205      * Backprojects a 2D conic into a 3D quadric.
206      * @param conic 2D conic to be backprojected.
207      * @return A backprojected 3D quadric.
208      */
209     public Quadric backProject(Conic conic) {
210         Quadric quadric = new Quadric();
211         backProject(conic, quadric);
212         return quadric;
213     }
214     
215     /**
216      * Backprojects a 2D conic into a 3D quadric and stores the result into
217      * provided instance.
218      * @param conic 2D conic to be backprojected.
219      * @param result Instance where data of backprojected 3D quadric will be
220      * stored.
221      */
222     public abstract void backProject(Conic conic, Quadric result);
223         
224     /**
225      * Projects a 3D dual quadric into a 2D dual conic.
226      * @param dualQuadric 3D dual quadric to be projected.
227      * @return A 2D dual conic.
228      */
229     public DualConic project(DualQuadric dualQuadric) {
230         DualConic dualConic = new DualConic();
231         project(dualQuadric, dualConic);
232         return dualConic;
233     }
234     
235     /**
236      * Projects a 3D dual quadric into a 2D dual conic and stores the result
237      * into provided instance.
238      * @param dualQuadric 3D dual quadric to be projected.
239      * @param result Instance where data of projected 2D dual conic will be
240      * stored.
241      */
242     public abstract void project(DualQuadric dualQuadric, DualConic result);
243     
244     /**
245      * Projects a 3D quadric into a 2D conic.
246      * The internal implementation of this method needs to compute the dual
247      * quadric of provided quadric, and also needs to compute the resulting
248      * conic from an internal dual conic, for that reason a CameraExceptio might 
249      * be raised if provided quadric is degenerate or if internal estimated dual 
250      * conic is degenerate and cannot be converted into a conic.
251      * @param quadric 3D quadric to be projected.
252      * @return A projected 2D conic.
253      * @throws CameraException thrown if there are geometric degeneracies.
254      */
255     public Conic project(Quadric quadric) throws CameraException {
256         Conic conic = new Conic();
257         project(quadric, conic);
258         return conic;
259     }
260     
261     /**
262      * Projects a 3D quadric into a 2D conic and stores the result into provided
263      * conic instance.
264      * The internal implementation of this method needs to compute the dual
265      * quadric of provided quadric, and also needs to compute the resulting
266      * conic from an internal dual conic, for that reason a 
267      * CameraException might be raised if provided quadric is degenerate or a 
268      * if internal estimated dual conic is degenerate and cannot be converted 
269      * into a conic.
270      * @param quadric 3D quadric to be projected.
271      * @param result Instance where data of projected 2D conic will be stored.
272      * @throws CameraException thrown if there are geometric degeneracies.
273      */
274     public void project(Quadric quadric, Conic result) throws CameraException {
275         quadric.normalize();
276         try {
277             DualQuadric dualQuadric = quadric.getDualQuadric();
278             DualConic dualConic = new DualConic();
279             project(dualQuadric, dualConic);
280             dualConic.conic(result);
281         } catch (GeometryException e) {
282             throw new CameraException(e);
283         }
284     }
285     
286     /**
287      * Backprojects a 2D dual conic into a 3D dual quadric.
288      * The internal implementation of this method needs to compute a conic from
289      * provided dual conic, and also needs to compute the resulting dual quadric
290      * from an internal quadric. For that reason a CameraException might be 
291      * raised if provided dual conic is degenerate or a if internal estimated
292      * quadric is degenerate and cannot be converted into a dual quadric.
293      * @param dualConic 2D dual conic to be backprojected.
294      * @return A backprojected 3D dual quadric.
295      * @throws CameraException thrown if there are geometric degeneracies.
296      */
297     public DualQuadric backProject(DualConic dualConic) throws CameraException {
298         DualQuadric dualQuadric = new DualQuadric();
299         backProject(dualConic, dualQuadric);
300         return dualQuadric;
301     }
302     
303     /**
304      * Backprojects a 2D dual conic into a 3D dual quadric and stores the result
305      * into provided dual quadric instance.
306      * The internal implementation of this method needs to compute a conic from
307      * provided dual conic, and also needs to compute the resulting dual quadric
308      * from an internal quadric. For that reason a CameraException might be 
309      * raised if provided dual conic is degenerate or a if internal estimated
310      * quadric is degenerate and cannot be converted into a dual quadric.
311      * @param dualConic 2D dual conic to be backprojected.
312      * @param result Instance where data of backprojected 3D dual quadric will
313      * be stored.
314      * @throws CameraException thrown if there are geometric degeneracies.
315      */
316     public void backProject(DualConic dualConic, DualQuadric result) 
317             throws CameraException {
318         try {
319             dualConic.normalize();
320             Conic conic = dualConic.getConic();
321             Quadric quadric = new Quadric();
322             backProject(conic, quadric);
323             quadric.dualQuadric(result);
324         } catch (GeometryException e) {
325             throw new CameraException(e);
326         }
327     }

329     /**
330      * Returns the type of this camera.
331      * @return Type of this camera.
332      */
333     public abstract CameraType getType();
334     
335     /**
336      * Creates an instance of a camera using default type.
337      * @return A newly instantiated camera.
338      */
339     public static Camera create() {
340         return create(DEFAULT_CAMERA_TYPE);
341     }
342     
343     /**
344      * Creates an instance of a camera using provided type.
345      * @param type Type of camera to be created.
346      * @return A newly instantiated camera.
347      */
348     public static Camera create(CameraType type) {
349         switch (type) {
350             case PINHOLE_CAMERA:
351             default:
352                 return new PinholeCamera();
353         }
354     }
