17  
18  import com.irurueta.algebra.AlgebraException;
19  import com.irurueta.algebra.Matrix;
20  import com.irurueta.algebra.SingularValueDecomposer;
21  import com.irurueta.algebra.Utils;

23  import java.io.Serializable;

25  /**
26   * This class defines a lines in 3D space.
27   * A line in 3D space is defined as the intersection of two non-parallel planes.
28   */
29  @SuppressWarnings("WeakerAccess")
30  public class Line3D implements Serializable {
31      
32      /**
33       * Positive threshold determine whether points lay inside (is locus) of a 
34       * given line or not.
35       */
36      public static final double DEFAULT_LOCUS_THRESHOLD = 1e-12;
37  
38      /**
39       * Minimum allowed threshold
40       */
41      public static final double MIN_THRESHOLD = 0.0;
42  
43      /**
44       * Constant defining the size of vector that define the direction of a line
45       */
46      private static final int INHOM_VECTOR_SIZE = 3;
47  
48      /**
49       * 1st plane forming this 3D line.
50       */
51      private Plane mPlane1;
52      
53      /**
54       * 2nd plane forming this 3D line.
55       */
56      private Plane mPlane2;
57      
58      /**
59       * Constructor.
60       * Sets planes intersecting into this 3D line.
61       * @param plane1 1st plane.
62       * @param plane2 2nd plane.
63       * @throws CoincidentPlanesException Raised if provided planes are 
64       * coincident and hence never intersect. Notice that parallel planes are
65       * not coincident and they intersect at infinity.
66       */
67      public Line3D(Plane plane1, Plane plane2) throws CoincidentPlanesException {
68          setPlanes(plane1, plane2);
69      }
70      
71      /**
72       * Constructor.
73       * Builds a 3D line passing through provided 2 points.
74       * @param point1 1st point.
75       * @param point2 2nd point.
76       * @throws CoincidentPointsException Raised if provided points are 
77       * considered to be equal.
78       */
79      public Line3D(Point3D point1, Point3D point2) 
80              throws CoincidentPointsException {
81          setPlanesFromPoints(point1, point2);
82      }
83      
84      /**
85       * Determines whether provided planes are coincident.
86       * Two planes are considered coincident if they are equal up to scale.
87       * @param plane1 1st plane.
88       * @param plane2 2nd plane.
89       * @return True if planes are coincident, false otherwise.
90       */
91      public static boolean areCoincidentPlanes(Plane plane1, Plane plane2) {
92          //normalize planes to increase accuracy
93          plane1.normalize();
94          plane2.normalize();
95          
96          try {
97              Matrix m = new Matrix(2, Plane.PLANE_NUMBER_PARAMS);
98              m.setElementAt(0, 0, plane1.getA());
99              m.setElementAt(0, 1, plane1.getB());
100             m.setElementAt(0, 2, plane1.getC());
101             m.setElementAt(0, 3, plane1.getD());
102         
103             m.setElementAt(1, 0, plane2.getA());
104             m.setElementAt(1, 1, plane2.getB());
105             m.setElementAt(1, 2, plane2.getC());
106             m.setElementAt(1, 3, plane2.getD());
107         
108             //check that matrix rank is 2 (and its nullity is also 2), in that 
109             //case its right null-space has dimension two, which is linear 
110             //combination of two 3D points (a line in 3D), and line is defined.
111             //Note that rank cannot be greater than 2 because matrix m has only 
112             //2 rows.
113             //If rank is smaller than 2, then the 2 planes are parallel, and the
114             //null-space has at least dimension 3, which is a perpendicular 
115             //plane or the whole space (Depending if nullity is 3 or 4)
116             return Utils.rank(m) < 2;
117         } catch (AlgebraException e) {
118             //if for numerical reasons it cannot be determined whether planes
119             //are parallel, it will be assumed the worst case, which is that
120             //planes are parallel, which is something common for very large
121             //values or planes close at infinity
122             return true; 
123         }
124     }
125     
126     /**
127      * Sets intersecting planes for this 3D line.
128      * The intersection of provided planes will determine this 3D line.
129      * @param plane1 1st plane.
130      * @param plane2 2nd plane.
131      * @throws CoincidentPlanesException Raised if provided planes are 
132      * coincident. Notice that parallel planes are not coincident and they
133      * intersect at infinity.
134      */
135     public final void setPlanes(Plane plane1, Plane plane2) 
136             throws CoincidentPlanesException {
137         if (areCoincidentPlanes(plane1, plane2)) {
138             throw new CoincidentPlanesException();
139         }
140         
141         mPlane1 = plane1;
142         mPlane2 = plane2;
143     }
144     
145     /**
146      * Sets planes of this 3D line so that it passes through provided points.
147      * @param point1 1st point.
148      * @param point2 2nd point.
149      * @throws CoincidentPointsException Raised if provided points are equal
150      * and hence a line cannot be determined.
151      */
152     public final void setPlanesFromPoints(Point3D point1, Point3D point2) 
153             throws CoincidentPointsException {
154         //build matrix containing director vector on a row
155         try {
156             Matrix m = new Matrix(1, INHOM_VECTOR_SIZE);
157             m.setElementAt(0, 0, point2.getInhomX() - point1.getInhomX());
158             m.setElementAt(0, 1, point2.getInhomY() - point1.getInhomY());
159             m.setElementAt(0, 2, point2.getInhomZ() - point1.getInhomZ());
160 
161             //m matrix will have rank 1 and nullity 2. The null-space will be 
162             //formed by two vectors perpendicular to the director vector        
163             SingularValueDecomposer decomposer = new SingularValueDecomposer(m);
164             decomposer.decompose();
165         
166             //check that points are not coincident, and hence all the values of 
167             //matrix m are not zero
168             if (decomposer.getRank() < 1) {
169                 throw new CoincidentPointsException();
170             }
171         
172             //last two columns of V contains director vectors of plane1 and 
173             //plane2
174             Matrix v = decomposer.getV();
175         
176             double[] directorVector1 = v.getSubmatrixAsArray(0, 1, 2, 1);
177             double[] directorVector2 = v.getSubmatrixAsArray(0, 2, 2, 2);
178         
179             mPlane1 = new Plane(point1, directorVector1);
180             mPlane2 = new Plane(point1, directorVector2);
181         } catch (AlgebraException e) {
182             throw new CoincidentPointsException(e);
183         }
184     }

186     /**
187      * Returns 1st plane determining this 3D line.
188      * The intersection of plane1 and plane2 determines this 3D line.
189      * @return 1st plane.
190      */
191     public Plane getPlane1() {
192         return mPlane1;
193     }
194     
195     /**
196      * Returns 2nd plane determining this 3D line.
197      * The intersection of plane1 and plane2 determines this 3D line.
198      * @return 2nd plane.
199      */
200     public Plane getPlane2() {
201         return mPlane2;
202     }
203     
204     /**
205      * Determines if provided point is locus of this 3D line up to provided
206      * threshold.
207      * @param point Point to be checked.
208      * @param threshold Threshold to determine if provided point is locus or 
209      * not. This should usually be a small value.
210      * @return True if provided point belongs to this 3D line, false otherwise.
211      * @throws IllegalArgumentException Raised if provided threshold is negative.
212      */
213     public boolean isLocus(Point3D point, double threshold) {
214         return mPlane1.isLocus(point, threshold) && 
215                 mPlane2.isLocus(point, threshold);
216     }
217     
218     /**
219      * Raised if provided point is locus of this 3D line.
220      * @param point Point to be checked.
221      * @return True if provided point belongs to this 3D line, false otherwise.
222      */
223     public boolean isLocus(Point3D point) {
224         return isLocus(point, DEFAULT_LOCUS_THRESHOLD);
225     }   
226     
227     /**
228      * Returns shortest distance of provided point to this 3D line.
229      * The shortest distance is obtained in perpendicular direction of this
230      * line.
231      * @param point Point to be checked.
232      * @return Shortest distance of provided point to this 3D line.
233      */
234     public double getDistance(Point3D point) {
235         Point3D closestPoint = getClosestPoint(point);
236         return point.distanceTo(closestPoint);
237     }
238     
239     /**
240      * Returns closest point belonging to this 3D line respect provided point.
241      * @param point Point to be checked.
242      * @return Closest point belonging to this 3D line respect provided point.
243      */
244     public Point3D getClosestPoint(Point3D point) {
245         return getClosestPoint(point, DEFAULT_LOCUS_THRESHOLD);
246     }
247     
248     /**
249      * Returns closest point belonging to this 3D line respect provided point
250      * up to provided threshold.
251      * @param point Point to be checked.
252      * @param threshold Threshold to determine closest point.
253      * @return Closest point belonging to this 3D line respect provided point.
254      * @throws IllegalArgumentException Raised if provided threshold is negative.
255      */
256     public Point3D getClosestPoint(Point3D point, double threshold) {
257         Point3D result = Point3D.create();
258         closestPoint(point, result, threshold);
259         return result;
260     }
261     
262     /**
263      * Computes closest point belonging to this 3D line respect provided point
264      * and stores the result in provided instance.
265      * @param point Point to be checked.
266      * @param result Instance where computed point will be stored.
267      */
268     public void closestPoint(Point3D point, Point3D result) {
269         closestPoint(point, result, DEFAULT_LOCUS_THRESHOLD);
270     }
271     
272     /**
273      * Computes closest point belonging to this 3D line respect provided point
274      * up to provided threshold and stores the result in provided instance.
275      * @param point Point to be checked.
276      * @param result Instance where computed point will be stored.
277      * @param threshold Threshold to determine closest point.
278      * @throws IllegalArgumentException Raised if provided threshold is negative.
279      */
280     public void closestPoint(Point3D point, Point3D result, double threshold) {
281         if (threshold < MIN_THRESHOLD) {
282             throw new IllegalArgumentException();
283         }
284         
285         //normalize to increase accuracy
286         point.normalize();
287         
288         //compute director vector perpendicular to director vectors of plane1 
289         //and plane2, and use it along provided point to set a 3rd plane.
290         //Using plane1, plane2 and the 3rd plane, we can get their intersection
291         //to obtain a point which will be locus of this line3 and will be 
292         //located at shortest distance of provided point to this line3.
293         
294         //This is a plane having as director vector this line 3D and passing 
295         //through provided point
296         Plane p = new Plane(point, getDirection());
297         try {
298             intersection(p, result);
299         } catch (NoIntersectionException ignore) {
300             //never happens
301         }
302     }
303     
304     /**
305      * Normalize the planes forming this 3D line.
306      */
307     public void normalize() {
308         mPlane1.normalize();
309         mPlane2.normalize();
310     }
311     
312     /**
313      * Determines whether the planes forming this 3D line are normalized or not.
314      * @return True if planes forming this 3D line are normalized, false 
315      * otherwise.
316      */
317     public boolean isNormalized() {
318         return mPlane1.isNormalized() && mPlane2.isNormalized();
319     }
320     
321     /**
322      * Returns array containing vector that indicates the direction of this 3D 
323      * line.
324      * @return Returns vector indicating the direction of this 3D line.
325      */
326     public double[] getDirection() {
327         try {
328             return Utils.crossProduct(mPlane1.getDirectorVector(), 
329                     mPlane2.getDirectorVector());
330         } catch (AlgebraException ignore) {
331             return null;
332         }
333     }
334     
335     /**
336      * Returns point where provided point intersects this 3D line.
337      * @param plane Plane to intersect this 3D line.
338      * @return Point where provided point intersects this 3D line.
339      * @throws NoIntersectionException Raised if provided plane does not 
340      * intersect this 3D line.
341      */
342     public Point3D getIntersection(Plane plane) throws NoIntersectionException {
343         Point3D result = Point3D.create();
344         intersection(plane, result);
345         return result;
346     }
347     
348     /**
349      * Computes point where provided point intersects this 3D line and stores
350      * the result in provided instance.
351      * @param plane Plane to intersect this 3D line.
352      * @param result Instance where computes point will be stored.
353      * @throws NoIntersectionException Raised if provided plane does not 
354      * intersect this 3D line.
355      */
356     public void intersection(Plane plane, Point3D result) 
357             throws NoIntersectionException {
358         //use plane1, plane2 and provided plane to find an intersection
359         plane.intersection(mPlane1, mPlane2, result);
360     }
