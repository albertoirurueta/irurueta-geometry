29       * lies inside the polygon or not, or if it's locus or not, etc.

158     /**
159      * Returns signed area of this polygon.
160      * The sign of the area determines whether vertices of the polygon are 
161      * provided in clockwise (negative sign) or clockwise (positive sign) order.
162      * @return Signed area of this polygon.
163      */
164     public double getArea() {
165         Point3D origin = mVertices.get(0);
166         double inhomX0 = origin.getInhomX();
167         double inhomY0 = origin.getInhomY();
168         double inhomZ0 = origin.getInhomZ();
169         
170         Iterator<Point3D> iterator = mVertices.iterator();
171         
172         Point3D prevPoint = iterator.next(); //because there are at least 3 
173                                             //vertices
174         Point3D curPoint;
175         double avgX = 0.0;
176         double avgY = 0.0;
177         double avgZ = 0.0;
178         while (iterator.hasNext()) {
179             curPoint = iterator.next();
180             
181             double inhomX1 = prevPoint.getInhomX();
182             double inhomY1 = prevPoint.getInhomY();
183             double inhomZ1 = prevPoint.getInhomZ();
184         
185             double inhomX2 = curPoint.getInhomX();
186             double inhomY2 = curPoint.getInhomY();
187             double inhomZ2 = curPoint.getInhomZ();
188         
189 
190             //compute cross product of ab = (prevPoint - origin) and 
191             //ac = (curPoint - origin)
192             double abX = inhomX1 - inhomX0;
193             double abY = inhomY1 - inhomY0;
194             double abZ = inhomZ1 - inhomZ0;
195         
196             double acX = inhomX2 - inhomX0;
197             double acY = inhomY2 - inhomY0;
198             double acZ = inhomZ2 - inhomZ0;
199         
200             double crossX = abY * acZ - abZ * acY;
201             double crossY = abZ * acX - abX * acZ;
202             double crossZ = abX * acY - abY * acX;
203 
204             avgX += crossX;
205             avgY += crossY;
206             avgZ += crossZ;
207                         
208             prevPoint = curPoint;
209         }
210         
211         return 0.5 * Math.sqrt(avgX * avgX + avgY * avgY + avgZ * avgZ);
212     }

214     /**
215      * Returns perimeter of this polygon.
216      * The perimeter is computed as the sum of the distances between consecutive
217      * pairs of vertices.
218      * @return Perimeter of this polygon.
219      */
220     public double getPerimeter() {
221         //iterate over all vertices and compute their distance
222         Iterator<Point3D> iterator = mVertices.iterator();
223         Point3D prevPoint = iterator.next();
224         Point3D point;
225         double perimeter = 0.0;
226         while (iterator.hasNext()) {
227             point = iterator.next();
228             perimeter += prevPoint.distanceTo(point);
229             prevPoint = point;
230         }
231         //get distance from last point with first one
232         perimeter += prevPoint.distanceTo(mVertices.get(0));
233         return perimeter;
234     }

237      * Determines if provided point lies within the region defined by this
238      * polygon.
239      * Notice that this method is only ensured to work for polygons having no
240      * holes or crossing borders. It will safely work on any other polygon,
241      * no matter if it is regular, non-regular, convex or concave.
242      * @param point Point to be checked.
243      * @return True if point lies within the area defined by this polygon, false
244      * otherwise.
245      * @throws TriangulatorException Raised if triangulation was required but
246      * failed.
247      */
248     public boolean isInside(Point3D point) throws TriangulatorException {
249         return isInside(point, DEFAULT_THRESHOLD);
250     }

252     /**
253      * Determines if provided point lies within the region defined by this
254      * polygon.
255      * Notice that this method is only ensured to work for polygons having no
256      * holes or crossing borders. It will safely work on any other polygon,
257      * no matter if it is regular, non-regular, convex or concave.
258      * @param point Point to be checked.
259      * @param threshold Threshold to determine whether point lies inside this
260      * polygon. Usually this value should be small.
261      * @return True if point lies within the area defined by this polygon, false
262      * otherwise.
263      * @throws IllegalArgumentException Raised if provided threshold is negative.
264      * @throws TriangulatorException Raised if triangulation was required but
265      * failed.
266      */
267     public boolean isInside(Point3D point, double threshold)
268             throws TriangulatorException {
269         if (threshold < MIN_THRESHOLD) {
270             throw new IllegalArgumentException();
271         }
272         
273         for (Triangle3D triangle : getTriangles()) {
274             if (triangle.isInside(point, threshold)) {
275                 return true;
276             }
277         }
278         return false;
279     }

293     /**
294      * Computes the center of this polygon.
295      * The center is the average point among all the vertices of this polygon.
296      * The center is not ensured to lie within the area formed by this polygon.
297      * @param result Instance where the computed center will be stored.
298      */
299     public void center(Point3D result) {
300         //compute average location of all vertices
301         double inhomX = 0.0;
302         double inhomY = 0.0;
303         double inhomZ = 0.0;
304         int total = mVertices.size();
305         
306         for (Point3D point : mVertices) {
307             inhomX += point.getInhomX() / (double)total;
308             inhomY += point.getInhomY() / (double)total;
309             inhomZ += point.getInhomZ() / (double)total;
310         }
311         result.setInhomogeneousCoordinates(inhomX, inhomY, inhomZ);
312     }

314     /**
315      * Determines whether provided point is locus of the borders defined by
316      * the vertices of this polygon. A point will be locus if it lies in the
317      * line defined by two consecutive vertices up to a certain threshold of
318      * error.
319      * @param point Point to be checked.
320      * @param threshold Threshold of allowed error. This should usually be a 
321      * small value.
322      * @return True if provided point lies in a border of this polygon, false
323      * otherwise.
324      * @throws IllegalArgumentException Raised if provided threshold is negative.
325      */
326     public boolean isLocus(Point3D point, double threshold) {
327         if (threshold < MIN_THRESHOLD) {
328             throw new IllegalArgumentException();
329         }
330         
331         //normalize point to increase accuracy
332         point.normalize();
333         
334         Iterator<Point3D> iterator = mVertices.iterator();
335         Point3D prevPoint = iterator.next(); //it's ok because there are at 
336                                             //least 3 vertices
337         prevPoint.normalize(); //normalize to increase accuracy
338         
339         Point3D curPoint;
340         while (iterator.hasNext()) {
341             curPoint = iterator.next();
342             curPoint.normalize(); //normalize to increase accuracy
343             if (point.isBetween(prevPoint, curPoint, threshold)) {
344                 return true;
345             }
346             prevPoint = curPoint;
347         }
348         
349         //check last point with first
350         return point.isBetween(prevPoint, mVertices.get(0), threshold);
351     }

365     /**
366      * Returns the shortest distance from provided point to a border of this
367      * polygon. Note that borders are segments defined by consecutive vertices
368      * @param point Point to be checked.
369      * @return Shortest distance from provided point to this polygon.
370      * @throws CoincidentPointsException Raised if points in a polygon are too
371      * close. This usually indicates numerical instability or polygon degeneracy.
372      */
373     public double getShortestDistance(Point3D point) 
374             throws CoincidentPointsException {
375         //iterate over all vertices and compute their distance
376         Iterator<Point3D> iterator = mVertices.iterator();
377         Point3D prevPoint = iterator.next();
378         prevPoint.normalize(); //to increase accuracy
379         Point3D curPoint;
380         double bestDist = Double.MAX_VALUE;
381         double dist;
382         boolean found = false;
383         Line3D line = null;
384         Point3D pointInLine = Point3D.create();
385         
386         while (iterator.hasNext()) {
387             curPoint = iterator.next();
388             curPoint.normalize(); //to increase accuracy
389             
390             //check if point lies in the segment of the boundary of this polygon
391             if (point.isBetween(curPoint, prevPoint)) {
392                 return 0.0;
393             }
394             
395             if (line == null) {
396                 line = new Line3D(curPoint, prevPoint);
397             } else {
398                 line.setPlanesFromPoints(curPoint, prevPoint);
399             }
400             line.normalize(); //to increase accuracy
401             
402             //find closest point to line
403             line.closestPoint(point, pointInLine);
404             pointInLine.normalize(); //to increase accuracy
405             
406             if (pointInLine.isBetween(curPoint, prevPoint)) {
407                 //closest point lies within segment of polygon boundary, so we
408                 //keep distance
409                 dist = point.distanceTo(pointInLine);
410                 if (dist < bestDist) {
411                     //a better point has been found
412                     bestDist = dist;
413                     found = true;
414                 }
415             }
416             
417             prevPoint = curPoint;
418         }
419         
420         //try last vertex with first
421         //check if point lies in the segment of the boundary of this polygon
422         Point3D first = mVertices.get(0);
423         if (point.isBetween(prevPoint, first)) {
424             return 0.0;
425         }
426             
427         if (line == null) {
428             line = new Line3D(prevPoint, first);
429         } else {
430             line.setPlanesFromPoints(prevPoint, first);
431         }
432         line.normalize(); //to increase accuracy
433             
434         //find closest point to line
435         line.closestPoint(point, pointInLine);
436         pointInLine.normalize(); //to increase accuracy
437             
438         if (pointInLine.isBetween(prevPoint, first)) {
439             //closest point lies within segment of polygon boundary, so we
440             //keep distance
441             dist = point.distanceTo(pointInLine);
442             if (dist < bestDist) {
443                 //a better point has been found
444                 bestDist = dist;
445                 found = true;
446             }
447         }
448         
449         
450         
451         if (!found) {
452             //no closest point was found on a segment belonging to polygon 
453             //boundary so we search for the closest vertex
454             iterator = mVertices.iterator();
455             while (iterator.hasNext()) {
456                 //a better vertex has been found
457                 curPoint = iterator.next();
458                 dist = point.distanceTo(curPoint);
459                 if (dist < bestDist) {
460                     bestDist = dist;
461                 }
462             }
463         }
464         
465         return bestDist;
466     }

483     /**
484      * Computes the closes point to provided point that is locus of this
485      * polygon (i.e. lies on a border of this polygon).
486      * @param point Point to be checked.
487      * @param result Instance where the closest point will be stored.
488      * @throws CoincidentPointsException Raised if points in a polygon are too
489      * close. This usually indicates numerical instability or polygon degeneracy.
490      */
491     public void closestPoint(Point3D point, Point3D result) 
492             throws CoincidentPointsException {
493         //iterate over all vertices and compute their distance
494         Iterator<Point3D> iterator = mVertices.iterator();
495         Point3D prevPoint = iterator.next();
496         prevPoint.normalize(); //to increase accuracy
497         
498         Point3D curPoint;
499         double bestDist = Double.MAX_VALUE;
500         double dist;
501         boolean found = false;
502         Line3D line = null;
503         Point3D pointInLine = Point3D.create();
504         
505         while (iterator.hasNext()) {
506             curPoint = iterator.next();
507             curPoint.normalize(); //to increase accuracy
508             
509             //check if point lies in the segment of the boundary of this polygon
510             if (point.isBetween(curPoint, prevPoint)) {
511                 result.setCoordinates(point);
512                 return;
513             }
514             
515             if (line == null) {
516                 line = new Line3D(curPoint, prevPoint);
517             } else {
518                 line.setPlanesFromPoints(curPoint, prevPoint);
519             }
520             line.normalize(); //to increase accuracy
521             
522             //find closest point to line
523             line.closestPoint(point, pointInLine);
524             pointInLine.normalize(); //to increase accuracy
525             
526             if (pointInLine.isBetween(curPoint, prevPoint)) {
527                 //closest point lies within segment of polygon boundary, so we
528                 //keep distance and point
529                 dist = point.distanceTo(pointInLine);
530                 if (dist < bestDist) {
531                     //a better point has been found
532                     bestDist = dist; 
533                     result.setCoordinates(pointInLine);
534                     found = true;
535                 }
536             }
537             
538             prevPoint = curPoint;
539         }
540         
541         //try last vertex with first
542         //check if point lies in the segment of the boundary of this polygon
543         Point3D first = mVertices.get(0);
544         if (point.isBetween(prevPoint, first)) {
545             result.setCoordinates(point);
546             return;
547         }
548             
549         if (line == null) {
550             line = new Line3D(prevPoint, first);
551         } else {
552             line.setPlanesFromPoints(prevPoint, first);
553         }
554         line.normalize(); //to increase accuracy
555             
556         //find closest point to line
557         line.closestPoint(point, pointInLine);
558         pointInLine.normalize(); //to increase accuracy
559             
560         if (pointInLine.isBetween(prevPoint, first)) {
561             //closest point lies within segment of polygon boundary, so we
562             //keep distance
563             dist = point.distanceTo(pointInLine);
564             if (dist < bestDist) {
565                 //a better point has been found
566                 bestDist = dist;
567                 result.setCoordinates(pointInLine);
568                 found = true;
569             }
570         }        
571         
572         
573         if (!found) {
574             //no closest point was found on a segment belonging to polygon 
575             //boundary so we search for the closest vertex
576             iterator = mVertices.iterator();
577             while (iterator.hasNext()) {
578                 curPoint = iterator.next();
579                 dist = point.distanceTo(curPoint);
580                 if (dist < bestDist) {
581                     //a better vertex has been found
582                     bestDist = dist;
583                     result.setCoordinates(curPoint);
584                 }
585             }
586         }
587     }

589     /**
590      * Triangulates this polygon using this polygon's triangulator method.
591      * A polygon only will be triangulated once when required or this method is
592      * called.
593      * This method will make no action if a polygon is already triangulated 
594      * unless it's vertices are reset.
595      * @throws TriangulatorException Raised if triangulation failed.
596      * @see #getTriangulatorMethod
597      * @see #setTriangulatorMethod(TriangulatorMethod)
598      */
599     public void triangulate() throws TriangulatorException {
600         if (!mTriangulated) {
601             Triangulator3D triangulator = Triangulator3D.create(
602                     mTriangulatorMethod);
603             mTriangles = triangulator.triangulate(mVertices);
604             mTriangulated = true;
605         }
606     }

608     /**
609      * Computes the average orientation of a 3D polygon as an array 
610      * containing the vector coordinates of the orientation.
611      * Notice that if all the vertices of the polygon lie in the same plane,
612      * then the estimated orientation will be exact, otherwise an average
613      * orientation will be estimated.
614      * Estimated orientation will be normalized (the norm of the estimated 
615      * vector will be 1).
616      * @param vertices List of vertices forming a polygon in 3D.
617      * @param result Array where the estimated orientation will be stored.
618      * @param threshold Threshold to determine whether the orientation can be
619      * estimated or not. Because the estimated orientation needs to be 
620      * normalized, it will only be possible to do so when norm has a reasonable
621      * value. If estimated norm happens to be smaller than provided value, then
622      * it will be assumed that polygon contains a degeneracy or that a 
623      * consecutive pair of vertices are coincident.
624      * @throws IllegalArgumentException Raised if provided result array does not
625      * have length 3, or if the list of provided vertices does not contain at
626      * least three vertices, or if provided threshold is negative.
627      * @throws CoincidentPointsException Raised usually when it is determined
628      * that consecutive vertices of the polygon are too close (i.e. coincident)
629      * or when there are polygon degeneracies or numerical instabilities.
630      */
631     public static void orientation(List<Point3D> vertices, double[] result,
632             double threshold) throws CoincidentPointsException {
633         int numVertices = vertices.size();
634         if (numVertices < MIN_VERTICES) {
635             throw new IllegalArgumentException();
636         }
637         if (result.length != INHOM_COORDS) {
638             throw new IllegalArgumentException();
639         }
640         if (threshold < MIN_THRESHOLD) {
641             throw new IllegalArgumentException();
642         }
643         
644         
645         double avgX = 0.0;
646         double avgY = 0.0;
647         double avgZ = 0.0;
648         
649         Iterator<Point3D> iterator = vertices.iterator();
650         Point3D prevPoint = iterator.next();
651         Point3D origin = prevPoint;
652         Point3D curPoint;
653         double inhomX0 = origin.getInhomX();
654         double inhomY0 = origin.getInhomY();
655         double inhomZ0 = origin.getInhomZ();
656         
657         
658         while (iterator.hasNext()) {
659             curPoint = iterator.next();
660             double inhomX1 = prevPoint.getInhomX();
661             double inhomY1 = prevPoint.getInhomY();
662             double inhomZ1 = prevPoint.getInhomZ();
663             
664             double inhomX2 = curPoint.getInhomX();
665             double inhomY2 = curPoint.getInhomY();
666             double inhomZ2 = curPoint.getInhomZ();
667             
668             //compute cross product of ab = (vertex1 - origin) and ac = (vertex2
669             //- origin)
670             double abX = inhomX1 - inhomX0;
671             double abY = inhomY1 - inhomY0;
672             double abZ = inhomZ1 - inhomZ0;
673         
674             double acX = inhomX2 - inhomX0;
675             double acY = inhomY2 - inhomY0;
676             double acZ = inhomZ2 - inhomZ0;
677         
678             double crossX = abY * acZ - abZ * acY;
679             double crossY = abZ * acX - abX * acZ;
680             double crossZ = abX * acY - abY * acX;
681             
682             avgX += crossX;
683             avgY += crossY;
684             avgZ += crossZ;
685             
686             prevPoint = curPoint;
687         }
688 
689         double norm = Math.sqrt(avgX * avgX + avgY * avgY + avgZ * avgZ);
690         
691         if (norm < threshold) {
692             throw new CoincidentPointsException();
693         }
694         
695         avgX /= norm;
696         avgY /= norm;
697         avgZ /= norm;
698         
699         result[0] = avgX;
700         result[1] = avgY;
701         result[2] = avgZ;        
702     }

704     /**
705      * Computes the average orientation of a 3D polygon as an array 
706      * containing the vector coordinates of the orientation.
707      * Notice that if all the vertices of the polygon lie in the same plane,
708      * then the estimated orientation will be exact, otherwise an average
709      * orientation will be estimated.
710      * Estimated orientation will be normalized (the norm of the estimated 
711      * vector will be 1).
712      * @param vertices List of vertices forming a polygon in 3D.
713      * @param result Array where the estimated orientation will be stored.
714      * @throws IllegalArgumentException Raised if provided result array does not
715      * have length 3, or if the list of provided vertices does not contain at

1051     /**
1052      * Internal method to compute polygon orientation from their respective
1053      * orientation vectors.
1054      * Orientation vectors are provided as arrays and can be obtained by calling
1055      * orientation(Polygon3D, double[]) among other methods.
1056      * @param orientation1 Orientation of 1st polygon.
1057      * @param orientation2 Orientation of 2nd polygon.
1058      * @return Angle between two polygons in radians.
1059      * @throws IllegalArgumentException Raised if any of the orientation arrays
1060      * do not have length 3.
1061      */
1062     private static double getAngleBetweenOrientations(double[] orientation1, 
1063             double[] orientation2) {
1064         if (orientation1.length != INHOM_COORDS ||
1065                 orientation2.length != INHOM_COORDS) {
1066             throw new IllegalArgumentException();
1067         }
1068         
1069         double x1 = orientation1[0];
1070         double y1 = orientation1[1];
1071         double z1 = orientation1[2];
1072         
1073         double x2 = orientation2[0];
1074         double y2 = orientation2[1];
1075         double z2 = orientation2[2];
1076         
1077         double norm1 = Math.sqrt(x1 * x1 + y1 * y1 + z1 * z1);
1078         double norm2 = Math.sqrt(x2 * x2 + y2 * y2 + z2 * z2);
1079         
1080         double dotProduct = (x1 * x2 + y1 * y2 + z1 * z2) / (norm1 * norm2);
1081         
1082         return Math.acos(dotProduct);        
1083     }
