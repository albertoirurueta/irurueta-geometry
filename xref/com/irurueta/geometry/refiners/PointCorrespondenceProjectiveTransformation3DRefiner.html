24  import com.irurueta.numerical.EvaluationException;
25  import com.irurueta.numerical.GradientEstimator;
26  import com.irurueta.numerical.MultiDimensionFunctionEvaluatorListener;
27  import com.irurueta.numerical.fitting.LevenbergMarquardtMultiDimensionFitter;
28  import com.irurueta.numerical.fitting.LevenbergMarquardtMultiDimensionFunctionEvaluator;
29  import com.irurueta.numerical.robust.InliersData;

31  import java.util.BitSet;
32  import java.util.List;

34  /**
35   * A 3D projective transformation refiner using point correspondences.
36   * This class takes into account an initial estimation, inlier point matches
37   * and their residuals to find a solution that minimizes error of inliers in 
38   * LMSE terms.
39   * Typically a refiner is used by a robust estimator, however it can also be 
40   * useful in some other situations.
41   */
42  public class PointCorrespondenceProjectiveTransformation3DRefiner extends
43          ProjectiveTransformation3DRefiner<Point3D, Point3D> {
44      
45      /**
46       * Point to be reused when computing residuals.
47       */
48      private Point3D mResidualTestPoint = Point3D.create(
49              CoordinatesType.HOMOGENEOUS_COORDINATES);
50  
51      /**
52       * Constructor.
53       */
54      public PointCorrespondenceProjectiveTransformation3DRefiner() { }
55      
56      /**
57       * Constructor.
58       * @param initialEstimation initial estimation to be set.
59       * @param keepCovariance true if covariance of estimation must be kept after
60       * refinement, false otherwise.
61       * @param inliers set indicating which of the provided matches are inliers.
62       * @param residuals residuals for matched samples.
63       * @param numInliers number of inliers on initial estimation.
64       * @param samples1 1st set of paired samples.
65       * @param samples2 2nd set of paired samples.
66       * @param refinementStandardDeviation standard deviation used for 
67       * Levenberg-Marquardt fitting.
68       */
69      public PointCorrespondenceProjectiveTransformation3DRefiner(
70              ProjectiveTransformation3D initialEstimation, 
71              boolean keepCovariance, BitSet inliers, double[] residuals, 
72              int numInliers, List<Point3D> samples1, List<Point3D> samples2,
73              double refinementStandardDeviation) {
74          super(initialEstimation, keepCovariance, inliers, residuals, numInliers, 
75                  samples1, samples2, refinementStandardDeviation);
76      }
77      
78      /**
79       * Constructor.
80       * @param initialEstimation initial estimation to be set.
81       * @param keepCovariance true if covariance of estimation must be kept after
82       * refinement, false otherwise.
83       * @param inliersData inlier data, typically obtained from a robust 
84       * estimator.
85       * @param samples1 1st set of paired samples.
86       * @param samples2 2nd set of paired samples.
87       * @param refinementStandardDeviation standard deviation used for 
88       * Levenberg-Marquardt fitting.
89       */
90      public PointCorrespondenceProjectiveTransformation3DRefiner(
91              ProjectiveTransformation3D initialEstimation, 
92              boolean keepCovariance, InliersData inliersData, 
93              List<Point3D> samples1, List<Point3D> samples2, 
94              double refinementStandardDeviation) {
95          super(initialEstimation, keepCovariance, inliersData, samples1, 
96                  samples2, refinementStandardDeviation);
97      }
98      
99      /**
100      * Refines provided initial estimation.
101      * This method always sets a value into provided result instance regardless
102      * of the fact that error has actually improved in LMSE terms or not.
103      * @param result instance where refined estimation will be stored.
104      * @return true if result improves (error decreases) in LMSE terms respect 
105      * to initial estimation, false if no improvement has been achieved.
106      * @throws NotReadyException if not enough input data has been provided.
107      * @throws LockedException if estimator is locked because refinement is 
108      * already in progress.
109      * @throws RefinerException if refinement fails for some reason (e.g. unable
110      * to converge to a result).
111      */
112     @Override
113     public boolean refine(ProjectiveTransformation3D result) throws NotReadyException,
114             LockedException, RefinerException {
115         if (isLocked()) {
116             throw new LockedException();
117         }
118         if (!isReady()) {
119             throw new NotReadyException();
120         }
121         
122         mLocked = true;
123         
124         if (mListener != null) {
125             mListener.onRefineStart(this, mInitialEstimation);
126         }
127         
128         mInitialEstimation.normalize();
129         
130         double initialTotalResidual = totalResidual(mInitialEstimation);
131 
132         try {
133             final double[] initParams = new double[
134                     ProjectiveTransformation3D.HOM_COORDS * 
135                     ProjectiveTransformation3D.HOM_COORDS];
136             //copy values
137             System.arraycopy(mInitialEstimation.getT().getBuffer(), 0, 
138                     initParams, 0, initParams.length);
139 
140             //output values to be fitted/optimized will contain residuals
141             double[] y = new double[mNumInliers];
142             //input values will contain 2 sets of 2D points to compute residuals
143             final int nDims = 
144                     2 * Point3D.POINT3D_HOMOGENEOUS_COORDINATES_LENGTH;
145             Matrix x = new Matrix(mNumInliers, nDims);
146             int nSamples = mInliers.length();
147             int pos = 0;
148             Point3D inputPoint, outputPoint;
149             for (int i = 0; i < nSamples; i++) {
150                 if (mInliers.get(i)) {
151                     //sample is inlier
152                     inputPoint = mSamples1.get(i);
153                     outputPoint = mSamples2.get(i);
154                     inputPoint.normalize();
155                     outputPoint.normalize();
156                     x.setElementAt(pos, 0, inputPoint.getHomX());
157                     x.setElementAt(pos, 1, inputPoint.getHomY());
158                     x.setElementAt(pos, 2, inputPoint.getHomZ());
159                     x.setElementAt(pos, 3, inputPoint.getHomW());
160                     x.setElementAt(pos, 4, outputPoint.getHomX());
161                     x.setElementAt(pos, 5, outputPoint.getHomY());
162                     x.setElementAt(pos, 6, outputPoint.getHomZ());
163                     x.setElementAt(pos, 7, outputPoint.getHomW());
164                     
165                     y[pos] = mResiduals[i];
166                     pos++;
167                 }
168             }
169             
170             LevenbergMarquardtMultiDimensionFunctionEvaluator evaluator =
171                     new LevenbergMarquardtMultiDimensionFunctionEvaluator() {
172                         
173                 private Point3D mInputPoint = Point3D.create(
174                         CoordinatesType.HOMOGENEOUS_COORDINATES);
175                 
176                 private Point3D mOutputPoint = Point3D.create(
177                         CoordinatesType.HOMOGENEOUS_COORDINATES);
178                 
179                 private ProjectiveTransformation3D mTransformation =
180                         new ProjectiveTransformation3D();
181                 
182                 private GradientEstimator mGradientEstimator =
183                         new GradientEstimator(
184                                 new MultiDimensionFunctionEvaluatorListener() {
185                     @Override
186                     public double evaluate(double[] params) {
187                         //copy values
188                         System.arraycopy(params, 0, 
189                                 mTransformation.getT().getBuffer(), 0, 
190                                 params.length);
191                         
192                         return residual(mTransformation, mInputPoint, 
193                                 mOutputPoint);
194                     }
195                 });
196                         
197                 @Override
198                 public int getNumberOfDimensions() {
199                     return nDims;
200                 }
201 
202                 @Override
203                 public double[] createInitialParametersArray() {
204                     return initParams;
205                 }
206 
207                 @Override
208                 public double evaluate(int i, double[] point, double[] params, 
209                         double[] derivatives) throws EvaluationException {
210                     mInputPoint.setHomogeneousCoordinates(point[0], point[1], 
211                             point[2], point[3]);
212                     mOutputPoint.setHomogeneousCoordinates(point[4], point[5], 
213                             point[6], point[7]);
214                     
215                     //copy values
216                     System.arraycopy(params, 0, 
217                             mTransformation.getT().getBuffer(), 0, 
218                             params.length);
219                     
220                     double y = residual(mTransformation, mInputPoint, 
221                             mOutputPoint);
222                     mGradientEstimator.gradient(params, derivatives);
223                     
224                     return y;
225 
226                 }
227             };
228             
229             LevenbergMarquardtMultiDimensionFitter fitter =
230                     new LevenbergMarquardtMultiDimensionFitter(evaluator, x, y,
231                     getRefinementStandardDeviation());
232             
233             fitter.fit();
234             
235             //obtain estimated params
236             double[] params = fitter.getA();
237             
238             //update transformation
239                     
240             //copy values 
241             System.arraycopy(params, 0, result.getT().getBuffer(), 0, 
242                     params.length);
243             
244             if (mKeepCovariance) {
245                 //keep covariance
246                 mCovariance = fitter.getCovar();
247             }
248             
249             double finalTotalResidual = totalResidual(result);
250             boolean errorDecreased = finalTotalResidual < initialTotalResidual;
251             
252             if (mListener != null) {
253                 mListener.onRefineEnd(this, mInitialEstimation, result, 
254                         errorDecreased);
255             }
256             
257             return errorDecreased;
258             
259         } catch (Exception e) {
260             throw new RefinerException(e);
261         } finally {
262             mLocked = false;
263         }
264     }
265     
266     /**
267      * Computes the residual between the affine transformation and a pair of 
268      * matched points.
269      * @param transformation a transformation.
270      * @param inputPoint input 2D point.
271      * @param outputPoint output 2D point.
272      * @return residual.
273      */
274     private double residual(ProjectiveTransformation3D transformation, 
275             Point3D inputPoint, Point3D outputPoint) {
276         inputPoint.normalize();
277         outputPoint.normalize();
278         
279         transformation.transform(inputPoint, mResidualTestPoint);        
280         return mResidualTestPoint.distanceTo(outputPoint);
281     }
282     
283     /**
284      * Computes total residual among all provided inlier samples.
285      * @param transformation a transformation.
286      * @return total residual.
287      */
288     private double totalResidual(ProjectiveTransformation3D transformation) {
289         double result = 0.0;
290         
291         int nSamples = mInliers.length();
292         Point3D inputPoint, outputPoint;
293         for (int i = 0; i < nSamples; i++) {
294             if (mInliers.get(i)) {
295                 //sample is inlier
296                 inputPoint = mSamples1.get(i);
297                 outputPoint = mSamples2.get(i);
298                 inputPoint.normalize();
299                 outputPoint.normalize();
300                 result += residual(transformation, inputPoint, outputPoint);
301             }
302         }
303         
304         return result;
305     }
