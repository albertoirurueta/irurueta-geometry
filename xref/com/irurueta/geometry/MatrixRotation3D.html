23  /**
24   * This class defines the amount of rotation for 3D points or planes.
25   * Rotation is defined internally as a matrix.
26   */
27  @SuppressWarnings("WeakerAccess")
28  public class MatrixRotation3D extends Rotation3D implements Serializable {
29      
30      /**
31       * Constant defining the number of rows on a 3D rotation matrix expressed
32       * in inhomogeneous coordinates.
33       */
34      public static final int ROTATION3D_INHOM_MATRIX_ROWS = 3;
35      
36      /**
37       * Constant defining the number of columns on a 3D rotation matrix expressed
38       * in inhomogeneous coordinates.
39       */
40      public static final int ROTATION3D_INHOM_MATRIX_COLS = 3;
41      
42      /**
43       * Constant defining the number of rows on a 3D rotation matrix expressed
44       * in homogeneous coordinates.
45       */
46      public static final int ROTATION3D_HOM_MATRIX_ROWS = 4;
47      
48      /**
49       * Constant defining the number of columns on a 3D rotation matrix expressed
50       * in homogeneous coordinates.
51       */
52      public static final int ROTATION3D_HOM_MATRIX_COLS = 4;
53      
54      /**
55       * Threshold to determine that a gimbal locked might have been achieved 
56       * when trying to find roll, pitch and yaw angles.
57       */
58      public static final double GIMBAL_THRESHOLD = 1e-6;
59          
60      /**
61       * Internal matrix containing rotation using inhomogeneous coordinates.
62       * This matrix will be square, 3x3, orthogonal and will have determinant
63       * equal to one.
64       */
65      protected Matrix internalMatrix;
66      
67      
68      /**
69       * Empty Constructor.
70       * Initializes rotation so that no rotation exists (i.e. internal matrix is
71       * the identity.
72       */
73      public MatrixRotation3D() {
74          try {
75              internalMatrix = Matrix.identity(ROTATION3D_INHOM_MATRIX_ROWS, 
76                      ROTATION3D_INHOM_MATRIX_ROWS);
77          } catch (WrongSizeException ignore) {
78              //never happens
79          }
80      }

83       * Copy constructor.
84       * Copies provided rotation into this instance.
85       * @param rotation Instance to be copied.
86       */
87      public MatrixRotation3D(MatrixRotation3D rotation) {
88          internalMatrix = rotation.internalMatrix.clone();
89      }
90      
91      /**
92       * Copy constructor.
93       * Copies and converts provided rotation into this instance.
94       * @param rotation Instance to be copied.

96      public MatrixRotation3D(Rotation3D rotation) {
97          internalMatrix = rotation.asInhomogeneousMatrix();
98      }

103      * Provided matrix can be expressed in either homogeneous or inhomogeneous

107      * be DEFAULT_VALID_THRESHOLD.
108      * @param m Matrix to create rotation from.
109      * @throws InvalidRotationMatrixException Raised if provided matrix is not

118      * Constructor.
119      * Creates a 3D rotation using provided matrix.

121      * coordinates, and it must also be orthogonal up to provided threshold, and
122      * must have determinant equal to 1.
123      * @param m Matrix to create rotation from.
124      * @param threshold Threshold to determine whether matrix is orthonormal.
125      * @throws InvalidRotationMatrixException Raised if provided matrix is not

127      * @throws IllegalArgumentException Raised if provided threshold is 
128      * negative.
129      * {@link #isValidRotationMatrix(Matrix)}.

132             throws InvalidRotationMatrixException {
133         fromMatrix(m, threshold);

137      * Constructor.
138      * Creates a 3D rotation using provided Euler angles expressed in radians.

140      * @param betaEuler Beta Euler angle expressed in radians.
141      * @param gammaEuler Gamma Euler angle expressed in radians.
142      */
143     public MatrixRotation3D(double alphaEuler, double betaEuler, 
144             double gammaEuler) {

146     }

176         setAxisAndRotation(axisX, axisY, axisZ, theta);

188     /**
189      * Returns a copy of the internal matrix so that the internal matrix cannot
190      * be modified accidentally.
191      * Returned matrix will be 3x3, orthogonal and will have determinant equal
192      * to one.
193      * @return Internal matrix containing rotation of this instance.
194      */
195     public Matrix getInternalMatrix() {
196         return internalMatrix.clone();
197     }
198     
199     /**
200      * Sets the internal matrix of this rotation. Provided matrix must be 3x3
201      * and orthonormal (orthogonal with determinant equal to 1).
202      * @param internalMatrix Internal matrix to be set.
203      * @throws InvalidRotationMatrixException Raised if provided matrix is not
204      * 3x3 or it is not orthonormal.
205      */
206     public final void setInternalMatrix(Matrix internalMatrix) 
207             throws InvalidRotationMatrixException {
208         setInternalMatrix(internalMatrix, DEFAULT_VALID_THRESHOLD);
209     }
210     
211     /**
212      * Sets the internal matrix of this rotation. Provided matrix must be 3x3
213      * and orthonormal (orthogonal with determinant equal to 1) up to an error
214      * equal to provided threshold.
215      * @param m Internal matrix to be set.
216      * @param threshold Threshold to determine whether matrix is orthonormal or
217      * not.
218      * @throws InvalidRotationMatrixException Raised if provided matrix is not
219      * 3x3 or it is not orthonormal.
220      * @throws IllegalArgumentException Raised if provided threshold is 
221      * negative.
222      */
223     public final void setInternalMatrix(Matrix m, double threshold)
224             throws InvalidRotationMatrixException {
225         if (m.getRows() != ROTATION3D_INHOM_MATRIX_ROWS ||
226                 m.getColumns() != ROTATION3D_INHOM_MATRIX_COLS) {
227             throw new InvalidRotationMatrixException();
228         }
229         if (!isValidRotationMatrix(m, threshold)) {
230             throw new InvalidRotationMatrixException();
231         }
232         
233         internalMatrix = m;
234     }

236     /**
237      * Returns alpha euler angle within the range -pi and pi.
238      * @return Alpha euler angle expressed in radians.
239      */
240     public double getAlphaEulerAngle() {
241         return Math.atan2(-internalMatrix.getElementAt(2, 0), 
242                 internalMatrix.getElementAt(2, 2));
243     }
244     
245     /**
246      * Returns beta euler angle within the range -pi/2 and pi/2.
247      * @return Beta euler angle expressed in radians.
248      */
249     public double getBetaEulerAngle() {
250         return Math.asin(internalMatrix.getElementAt(2, 1));
251     }
252     
253     /**
254      * Returns gamma euler angle within the range -pi and pi.
255      * @return Gamma euler angle expressed in radians.
256      */
257     public double getGammaEulerAngle() {
258         return Math.atan2(-internalMatrix.getElementAt(0, 1), 
259                 internalMatrix.getElementAt(1, 1));
260     }
261     
262     /**
263      * Sets euler angles of this rotation, expressed in radians.
264      * @param alphaEuler Alpha euler angle in radians.
265      * @param betaEuler Beta euler angle in radians.
266      * @param gammaEuler Gamma euler angle in radians.
267      */
268     public final void setEulerAngles(double alphaEuler, double betaEuler, 
269             double gammaEuler) {
270         
271         double sinAlpha = Math.sin(alphaEuler);
272         double cosAlpha = Math.cos(alphaEuler);
273         
274         double sinBeta = Math.sin(betaEuler);
275         double cosBeta = Math.cos(betaEuler);
276         
277         double sinGamma = Math.sin(gammaEuler);
278         double cosGamma = Math.cos(gammaEuler);
279         
280         //reuse internal matrix if possible
281         try {
282             if (internalMatrix == null) {
283                 internalMatrix = new Matrix(ROTATION3D_INHOM_MATRIX_ROWS,
284                         ROTATION3D_INHOM_MATRIX_COLS);
285             }
286 
287             internalMatrix.setElementAt(0, 0, cosAlpha * cosGamma -
288                     sinAlpha * sinBeta * sinGamma);
289             internalMatrix.setElementAt(1, 0, cosAlpha * sinGamma +
290                     sinAlpha * sinBeta * cosGamma);
291             internalMatrix.setElementAt(2, 0, -sinAlpha * cosBeta);
292 
293             internalMatrix.setElementAt(0, 1, -cosBeta * sinGamma);
294             internalMatrix.setElementAt(1, 1, cosBeta * cosGamma);
295             internalMatrix.setElementAt(2, 1, sinBeta);
296 
297             internalMatrix.setElementAt(0, 2, sinAlpha * cosGamma +
298                     cosAlpha * sinBeta * sinGamma);
299             internalMatrix.setElementAt(1, 2, sinAlpha * sinGamma -
300                     cosAlpha * sinBeta * cosGamma);
301             internalMatrix.setElementAt(2, 2, cosAlpha * cosBeta);
302         } catch (WrongSizeException ignore) {
303             //never happens
304         }
305     }

307     /**
308      * Returns roll angle around x axis expressed in radians for the 1st 
309      * possible set of solutions.
310      * When obtaining roll, pitch and yaw angles from a rotation matrix, there 
311      * might be two possible sets of solutions (#getRollAngle(), 
312      * #getPitchAngle(), #getYawAngle()) or (#getRollAngle2(), 
313      * #getPitchAngle2(), #getYawAngle2()).
314      * @return roll angle around x axis.
315      * @see <a href="http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf">http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf</a>
316      * @see <a href="https://github.com/joansola/slamtb">R2e.m at https://github.com/joansola/slamtb</a>
317      */
318     public double getRollAngle() {
319         return getRollAngle(getPitchAngle());
320     }

322     /**
323      * Returns roll angle around x axis expressed in radians for the 2nd 
324      * possible set of solutions.
325      * When obtaining roll, pitch and yaw angles from a rotation matrix, there 
326      * might be two possible sets of solutions (#getRollAngle(), 
327      * #getPitchAngle(), #getYawAngle()) or (#getRollAngle2(), 
328      * #getPitchAngle2(), #getYawAngle2()).
329      * @return roll angle around x axis.
330      * @see <a href="http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf">http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf</a>
331      * @see <a href="https://github.com/joansola/slamtb">R2e.m at https://github.com/joansola/slamtb</a>
332      */
333     public double getRollAngle2() {
334         return getRollAngle(getPitchAngle2());
335     }

337     /**
338      * Returns roll angle around x axis expressed in radians corresponding to
339      * provided pitch value. 
340      * This method is used internally.
341      * @param pitch pitch angle expressed in radians.
342      * @return roll angle around x axis.
343      */
344     private double getRollAngle(double pitch) {
345         if (!hasGimbalLock()) {
346             double cosPitch = Math.cos(pitch);
347             return Math.atan2(internalMatrix.getElementAt(2, 1) / cosPitch, 
348                 internalMatrix.getElementAt(2, 2) / cosPitch);        
349         } else {
350             //gimbal lock (pitch is close to +-90 degrees)
351             if (internalMatrix.getElementAt(2, 0) < 0.0) {
352                 //pitch is +90 degrees
353                 return Math.atan2(internalMatrix.getElementAt(0, 1), 
354                         internalMatrix.getElementAt(0, 2));
355             } else {
356                 //pitch is -90 degrees
357                 return Math.atan2(-internalMatrix.getElementAt(0, 1),
358                         -internalMatrix.getElementAt(0, 2));
359             }
360         }                
361     }

363     /**
364      * Returns pitch angle around y axis expressed in radians for the 1st 
365      * possible set of solutions.
366      * When obtaining roll, pitch and yaw angles from a rotation matrix, there 
367      * might be two possible sets of solutions (#getRollAngle(), 
368      * #getPitchAngle(), #getYawAngle()) or (#getRollAngle2(), 
369      * #getPitchAngle2(), #getYawAngle2()).
370      * @return pitch angle around y axis.
371      * @see <a href="http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf">http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf</a>
372      * @see <a href="https://github.com/joansola/slamtb">R2e.m at https://github.com/joansola/slamtb</a>
373      */
374     public double getPitchAngle() {
375         return -Math.asin(internalMatrix.getElementAt(2, 0));
376     }
377     
378     /**
379      * Returns pitch angle around y axis expressed in radians for the 2nd
380      * possible set of solutions.
381      * When obtaining roll, pitch and yaw angles from a rotation matrix, there 
382      * might be two possible sets of solutions (#getRollAngle(), 
383      * #getPitchAngle(), #getYawAngle()) or (#getRollAngle2(), 
384      * #getPitchAngle2(), #getYawAngle2()).
385      * When a gimbal lock occurs this both pitch angles are equal because only
386      * yaw is undefined, buth pitch and roll are unique.
387      * @return pitch angle around y axis.
388      * @see <a href="http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf">http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf</a>
389      * @see <a href="https://github.com/joansola/slamtb">R2e.m at https://github.com/joansola/slamtb</a>
390      */
391     public double getPitchAngle2() {
392         if (!hasGimbalLock()) {
393             return Math.PI - getPitchAngle();
394         } else {
395             return getPitchAngle();
396         }
397     }

420      * When a gimbal lock occurs (pitch angle is close to +- 90 degrees), then
421      * yaw angle is undefined, and can be any value, although this method will
422      * return 0.0.
423      * When obtaining roll, pitch and yaw angles from a rotation matrix, there 
424      * might be two possible sets of solutions (#getRollAngle(), 
425      * #getPitchAngle(), #getYawAngle()) or (#getRollAngle2(), 
426      * #getPitchAngle2(), #getYawAngle2()).
427      * @return yaw angle around z axis.
428      * @see <a href="http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf">http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf</a>

435     /**
436      * Returns yaw angle around x axis expressed in radians corresponding to
437      * provided pitch value. 
438      * This method is used internally.
439      * @param pitch pitch angle expressed in radians.
440      * @return yaw angle around x axis.
441      */
442     private double getYawAngle(double pitch) {
443         if (!hasGimbalLock()) {
444             double cosPitch = Math.cos(pitch);
445             return Math.atan2(internalMatrix.getElementAt(1, 0) / cosPitch, 
446                     internalMatrix.getElementAt(0, 0) / cosPitch);
447         } else {
448             //gimbal lock (pitch is close to +-90 degrees)
449             return 0.0; //can be anything.
450         }        
451     }

453     /**
454      * Indicates whether current rotation contains ambiguities (a.k.a gimbal
455      * lock). This situation happens when pitch angle is close to +-90 degrees).
456      * @return true if current rotation contains a gimbal lock, false otherwise.
457      * @see <a href="http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf">http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf</a>
458      * @see <a href="https://github.com/joansola/slamtb">R2e.m at https://github.com/joansola/slamtb</a>
459      */
460     public boolean hasGimbalLock() {
461         return Math.abs(Math.abs(internalMatrix.getElementAt(2, 0)) - 1.0) < 
462                 GIMBAL_THRESHOLD;
463     }

465     /**
466      * Sets rotation angles, expressed in radians.
467      * @param roll roll angle in radians around x axis.
468      * @param pitch pitch angle in radians around y axis.
469      * @param yaw yaw angle in radians around z axis.
470      * @see <a href="http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf">http://www.staff.city.ac.uk/~sbbh653/publications/euler.pdf</a>
471      * @see <a href="https://github.com/joansola/slamtb">e2R.m at https://github.com/joansola/slamtb</a>
472      */
473     public void setRollPitchYaw(double roll, double pitch, double yaw) {
474         double sr = Math.sin(roll);
475         double cr = Math.cos(roll);
476         
477         double sp = Math.sin(pitch);
478         double cp = Math.cos(pitch);
479         
480         double sy = Math.sin(yaw);
481         double cy = Math.cos(yaw);
482 
483         try {
484             //reuse internal matrix if possible
485             if (internalMatrix == null) {
486                 internalMatrix = new Matrix(ROTATION3D_INHOM_MATRIX_ROWS,
487                         ROTATION3D_INHOM_MATRIX_COLS);
488             }
489 
490             internalMatrix.setElementAt(0, 0, cp * cy);
491             internalMatrix.setElementAt(1, 0, cp * sy);
492             internalMatrix.setElementAt(2, 0, -sp);
493 
494             internalMatrix.setElementAt(0, 1, -cr * sy + sr * sp * cy);
495             internalMatrix.setElementAt(1, 1, cr * cy + sr * sp * sy);
496             internalMatrix.setElementAt(2, 1, sr * cp);
497 
498             internalMatrix.setElementAt(0, 2, sr * sy + cr * sp * cy);
499             internalMatrix.setElementAt(1, 2, -sr * cy + cr * sp * sy);
500             internalMatrix.setElementAt(2, 2, cr * cp);
501         } catch (WrongSizeException ignore) {
502             //never happens
503         }
504     }

506     /**
507      * Sets the axis and rotation of this instance.
508      * Once set, points will rotate around provided axis an amount equal to
509      * provided rotation angle in radians.
510      * Note: to avoid numerical instabilities and improve accuracy, axis
511      * coordinates should be normalized (e.g. norm equal to 1).
512      * @param axisX X coordinate of rotation axis.
513      * @param axisY Y coordinate of rotation axis.
514      * @param axisZ Z coordinate of rotation axis.
515      * @param theta Amount of rotation in radians.
516      */
517     @Override
518     public final void setAxisAndRotation(double axisX, double axisY, 
519             double axisZ, double theta) {
520         double axisX2 = axisX * axisX;
521         double axisY2 = axisY * axisY;
522         double axisZ2 = axisZ * axisZ;
523         
524         double axisXY = axisX * axisY;
525         double axisXZ = axisX * axisZ;
526         double axisYZ = axisY * axisZ;
527         
528         double sinTheta = Math.sin(theta);
529         double cosTheta = Math.cos(theta);
530 
531         try {
532             if (internalMatrix == null) {
533                 internalMatrix = new Matrix(ROTATION3D_INHOM_MATRIX_ROWS,
534                         ROTATION3D_INHOM_MATRIX_COLS);
535             }
536 
537             internalMatrix.setElementAt(0, 0, axisX2 + (1.0 - axisX2) * cosTheta);
538             internalMatrix.setElementAt(1, 0, axisXY * (1.0 - cosTheta) +
539                     axisZ * sinTheta);
540             internalMatrix.setElementAt(2, 0, axisXZ * (1.0 - cosTheta) -
541                     axisY * sinTheta);
542 
543             internalMatrix.setElementAt(0, 1, axisXY * (1.0 - cosTheta) -
544                     axisZ * sinTheta);
545             internalMatrix.setElementAt(1, 1, axisY2 + (1.0 - axisY2) * cosTheta);
546             internalMatrix.setElementAt(2, 1, axisYZ * (1.0 - cosTheta) +
547                     axisX * sinTheta);
548 
549             internalMatrix.setElementAt(0, 2, axisXZ * (1.0 - cosTheta) +
550                     axisY * sinTheta);
551             internalMatrix.setElementAt(1, 2, axisYZ * (1.0 - cosTheta) -
552                     axisX * sinTheta);
553             internalMatrix.setElementAt(2, 2, axisZ2 + (1.0 - axisZ2) * cosTheta);
554         } catch (WrongSizeException ignore) {
555             //never happens
556         }
557     }

559     /**
560      * Returns rotation axis corresponding to this instance.
561      * Result is stored in provided axis array, which must have length 3.
562      * @param axis Array where axis coordinates will be stored.
563      * @throws IllegalArgumentException Raised if provided array does not have
564      * length 3.
565      * @throws RotationException Raised if numerical instabilities happen.
566      * Because internal matrix will always be well defined (orthogonal and
567      * determinant equal to 1), this exception will rarely happen.
568      */
569     @Override
570     public void rotationAxis(double[] axis) throws RotationException {
571         if (axis.length != ROTATION3D_INHOM_MATRIX_ROWS) {
572             throw new IllegalArgumentException();
573         }
574 
575         try {
576             //Rotation axis follows: R*v = v, which means that rotation axis is
577             //left unchanged after rotation. Hence:
578             //R*v = I*v --> (R - I)*v = 0, consequently rotation axis is the
579             //null-space of R-I
580             Matrix identity = Matrix.identity(ROTATION3D_INHOM_MATRIX_ROWS, 
581                     ROTATION3D_INHOM_MATRIX_COLS);
582             //line below: identity = internalMatrix - identity
583             internalMatrix.subtract(identity, identity);
584             //internalMatrix - identity
585         
586             SingularValueDecomposer decomposer = new SingularValueDecomposer(
587                     identity);
588         
589             decomposer.decompose();
590         
591             Matrix v = decomposer.getV();
592         
593             //last column of V contains axis values
594             axis[0] = v.getElementAt(0, 2);
595             axis[1] = v.getElementAt(1, 2);
596             axis[2] = v.getElementAt(2, 2);
597         } catch (AlgebraException e) {
598             throw new RotationException(e);
599         }        
600     }

602     /**
603      * Returns rotation amount or angle in radians around the rotation axis
604      * associated to this instance.
605      * @return Rotation angle in radians.
606      * @throws RotationException Raised if numerical instabilities happen.
607      * Because internal matrix will always be well defined (orthogonal and
608      * determinant equal to 1), this exception will rarely happen.
609      */
610     @Override
611     public double getRotationAngle() throws RotationException {
612         
613         //obtain rotation axis
614         double[] axis = getRotationAxis();
615         
616         try {
617             Matrix axisMatrix = new Matrix(1, ROTATION3D_INHOM_MATRIX_COLS);
618             axisMatrix.setSubmatrix(0, 0, 0,
619                     ROTATION3D_INHOM_MATRIX_COLS - 1, axis);
620         
621             SingularValueDecomposer decomposer = 
622                     new SingularValueDecomposer(axisMatrix);
623             decomposer.decompose();
624         
625             Matrix v = decomposer.getV();
626         
627             //because axisMatrix has rank 1, its null-space will contain a
628             //two-dimensional space (two vectors) perpendicular to axisMatrix
629             Matrix perpendicular = v.getSubmatrix(0, 2, 2, 2);
630             double normPerpendicular = Utils.normF(perpendicular);
631         
632             //use internal matrix to rotate perpendicular vector
633             Matrix rotated = internalMatrix.multiplyAndReturnNew(perpendicular);
634             double normRotated = Utils.normF(rotated);
635         
636             //normalize vectors
637             perpendicular.multiplyByScalar(1.0 / normPerpendicular);
638             rotated.multiplyByScalar(1.0 / normRotated);
639         
640             //their dot product is the cosine of their angle
641             //(we transpose rotated matrix to compute dot product)
642             rotated.transpose();
643             rotated.multiply(perpendicular);
644             double dotProduct = rotated.getElementAtIndex(0);
645         
646             double theta = Math.acos(dotProduct);
647         
648             //we need to determine sign of theta, for that reason we instantiate 
649             //two camera rotations with theta and -theta using the same rotation 
650             //axis and check for the rotation matrix that produces less error
651             MatrixRotation3D rotation1 = new MatrixRotation3D(axis, theta);
652             MatrixRotation3D rotation2 = new MatrixRotation3D(axis, -theta);
653         
654             Matrix rotationMatrix1 = rotation1.internalMatrix;
655             Matrix rotationMatrix2 = rotation2.internalMatrix;
656         
657             //normalize rotation matrices for their comparison
658             double norm1 = Utils.normF(rotationMatrix1);
659             double norm2 = Utils.normF(rotationMatrix2);
660             rotationMatrix1.multiplyByScalar(1.0 / norm1);
661             rotationMatrix2.multiplyByScalar(1.0 / norm2);
662         
663             //normalize current internal matrix for its comparison
664             double internalNorm = Utils.normF(internalMatrix);
665             Matrix internal = internalMatrix.multiplyByScalarAndReturnNew(
666                     1.0 / internalNorm);
667         
668             //compare matrices
669             rotationMatrix1.subtract(internal);
670             rotationMatrix2.subtract(internal);
671         
672             //now rotation matrices 1 and 2 contain the difference
673 
674             double normDiff1 = Utils.normF(rotationMatrix1);
675             double normDiff2 = Utils.normF(rotationMatrix2);
676         
677             //pich the rotation matrix that produces less error
678             if (normDiff1 < normDiff2) {
679                 return theta; //positive theta
680             } else {
681                 return -theta; //negative theta
682             }
683         } catch (AlgebraException e) {
684             throw new RotationException(e);
685         }
686     }

817      * Sets amount of rotation from provided inhomogeneous rotation matrix.
818      * Provided matrix must be orthogonal (i.e. squared, non-singular, it's
819      * transpose must be it's inverse) and must have determinant equal to 1.
820      * Provided matrix must also have size 3x3.
821      * @param m Provided rotation matrix.
822      * @param threshold Threshold to determine whether matrix is orthonormal.
823      * @throws InvalidRotationMatrixException Raised if provided matrix is not
824      * valid (has wrong size or it is not orthonormal).
825      * @throws IllegalArgumentException Raised if provided threshold is 
826      * negative.
827      * {@link #isValidRotationMatrix(Matrix)}.
828      */    
829     @Override
830     public void fromInhomogeneousMatrix(Matrix m, double threshold)
831             throws InvalidRotationMatrixException {
832         setInternalMatrix(m, threshold);
833     }

835     /**
836      * Sets amount of rotation from provided homogeneous rotation matrix.
837      * Provided matrix must be orthogonal (i.e. squared, non-singular, it's
838      * transpose must be it's inverse) and must have determinant equal to 1.
839      * Provided matrix must also have size 4x4, and its last row and column must
840      * be zero, except for element in last row and column which must be 1.
841      * @param m Provided rotation matrix.
842      * @param threshold Threshold to determine whether matrix is orthonormal.
843      * @throws InvalidRotationMatrixException Raised if provided matrix is not
844      * valid (has wrong size or it is not orthonormal).
845      * @throws IllegalArgumentException Raised if provided threshold is 
846      * negative.
847      * {@link #isValidRotationMatrix(Matrix)}.
848      */      
849     @Override
850     public void fromHomogeneousMatrix(Matrix m, double threshold)
851             throws InvalidRotationMatrixException {
852         if (m.getRows() != ROTATION3D_HOM_MATRIX_ROWS ||
853                 m.getColumns() != ROTATION3D_HOM_MATRIX_COLS) {
854             throw new InvalidRotationMatrixException();
855         }
856         if (!isValidRotationMatrix(m, threshold)) {
857             throw new InvalidRotationMatrixException();
858         }
859         if (Math.abs(m.getElementAt(3, 0)) > threshold ||
860                 Math.abs(m.getElementAt(3, 1)) > threshold ||
861                 Math.abs(m.getElementAt(3, 2)) > threshold ||
862                 Math.abs(m.getElementAt(0, 3)) > threshold ||
863                 Math.abs(m.getElementAt(1, 3)) > threshold ||
864                 Math.abs(m.getElementAt(2, 3)) > threshold ||
865                 Math.abs(m.getElementAt(3, 3) - 1.0) > threshold) {
866             throw new InvalidRotationMatrixException();
867         }
868         
869         internalMatrix = m.getSubmatrix(0, 0, ROTATION3D_INHOM_MATRIX_ROWS - 1, 
870                 ROTATION3D_INHOM_MATRIX_COLS - 1);
871     }

873     /**
874      * Rotates a 3D point using the origin of coordinates as the axis of 
875      * rotation.
876      * Point will be rotated by the amount of rotation contained in this 
877      * instance.
878      * @param inputPoint Input point to be rotated.
879      * @param resultPoint Rotated point.
880      */
881     @Override
882     public void rotate(Point3D inputPoint, Point3D resultPoint) {
883         try {
884             Matrix r = asHomogeneousMatrix();
885             Matrix p = new Matrix(
886                 Point3D.POINT3D_HOMOGENEOUS_COORDINATES_LENGTH, 1);
887             
888             inputPoint.normalize(); //to increase accuracy
889             p.setElementAt(0, 0, inputPoint.getHomX());
890             p.setElementAt(1, 0, inputPoint.getHomY());
891             p.setElementAt(2, 0, inputPoint.getHomZ());
892             p.setElementAt(3, 0, inputPoint.getHomW());
893             
894             //Rotated point below is R * p
895             r.multiply(p);
896             
897             resultPoint.setHomogeneousCoordinates(r.getElementAt(0, 0),
898                     r.getElementAt(1, 0), r.getElementAt(2, 0),
899                     r.getElementAt(3, 0));
900         } catch (WrongSizeException ignore) {
901             //never happens
902         }
903     }
